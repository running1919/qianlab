00448  */
00449 static void TC0_IrqHandler(void)
00450 {
00451     Tc *pTc0 = TC0;
00452     sDmad *pDmad = &dmad;
00453     uint8_t size;
00454     uint32_t nextBuffer = 1 - usartCurrentBuffer;
00455     uint32_t dmaChannel = usartDmaRxChannel;
00456     uint8_t  iController = (dmaChannel >> 8);
00457     uint8_t  iChannel    = (dmaChannel) & 0xFF;
00458     volatile uint32_t status = pTc0->TC_CHANNEL[0].TC_SR;
00459 
00460     if ((status & TC_SR_CPCS) != 0)
00461     {
00462         /* Serial Timer */
00463         if (isSerialPortON)
00464         {
00465             /* Get Received size */
00466             size = DMAC_GetDestinationAddr(pDmad->pDmacs[iController], iChannel)
00467                     - (uint32_t)(usartBuffers[usartCurrentBuffer]);
00468             if (size == 0)
00469             {
00470                 pTc0->TC_CHANNEL[0].TC_CCR = TC_CCR_CLKEN | TC_CCR_SWTRG;
00471                 return;
00472             }
00473 
00474             /* Stop DMA */
00475             DMAD_StopTransfer(pDmad, dmaChannel);
00476             /* Restart DMA in next buffer */
00477             _UsartDmaRx( nextBuffer );
00478 
00479             /* Send current buffer through the USB */
00480             CDCDSerial_Write(usartBuffers[usartCurrentBuffer],
00481                              size, 0, 0);
00482             usartCurrentBuffer = nextBuffer;
00483             pTc0->TC_CHANNEL[0].TC_CCR = TC_CCR_CLKEN | TC_CCR_SWTRG;
00484         }
00485     }
