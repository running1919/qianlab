01593  */

01595                         void* pBuffer, uint8_t bBPP,
01596                         uint16_t wX, uint16_t wY,
01597                         uint16_t wW, uint16_t wH )
01598 {
01599     void* pOldBuffer;
01600     uint32_t maxW = BOARD_LCD_WIDTH;
01601     uint32_t maxH = BOARD_LCD_HEIGHT;
01602 
01603     uint32_t bitsPR, bytesPR;
01604 
01605     switch (bLayer)
01606     {
01607         case LCDD_BASE:
01608             wX = 0; wY = 0;
01609             break;
01610         case LCDD_OVR1: case LCDD_HEO:
01611             /* Size check */
01612             if (wX + wW > BOARD_LCD_WIDTH || wY + wH > BOARD_LCD_HEIGHT)
01613                 return NULL;

01618                 || wW > 128 || wH > 128)

01624     if (wH == 0) wH = maxH - wY;
01625 
01626     bitsPR = wW * bBPP;
01627     bytesPR = (bitsPR&0x7) ? (bitsPR/8 + 1) : (bitsPR/8);
01628     memset(pBuffer, 0xFF, bytesPR*wH);
