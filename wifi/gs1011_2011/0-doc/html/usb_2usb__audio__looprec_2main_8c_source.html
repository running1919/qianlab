00029 /** \cond usb_audio_looprec
00030  *  \page usb_audio_looprec USB Audio Loopback-Recorder Example
00031  *
00032  *  \section Purpose
00033  *
00034  *  The USB Audio Loopback-Recorder Example will help you to get
00035  *  familiar with the USB Device Port(UDP) and DACC on AT91SAM microcontrollers.
00036  *  Also it can help you to be familiar with the USB Framework that is used for
00037  *  rapid development of USB-compliant class drivers such as USB Audio Device
00038  *  class.
00039  *
00040  *  \section Requirements
00041  *
00042  *  This package can be used with SAM9XX5 evaluation kits that have both
00043  *  UDP.
00044  *
00045  *  \section Description
00046  *
00047  *  The demo simulates an USB Desktop Speaker with Microphone which actually
00048  *  does not "speak out" but loop back the sound as microphone input.
00049  *
00050  *  When an EK running this program connected to a host (PC for example), with
00051  *  USB cable, the EK appears as a desktop speaker for the host. Then the host
00052  *  can play sound through host software. The audio stream from the host is
00053  *  then sent to the EK. At the same time, the audio stream received is also
00054  *  sent back to host from EK for recording.
00055  *
00056  *  \section Usage
00057  *
00058  *  -# Build the program and download it inside the evaluation board. Please
00059  *     refer to the
00060  *     <a href="http://www.atmel.com/dyn/resources/prod_documents/doc6421.pdf">
00061  *     SAM-BA User Guide</a>, the
00062  *     <a href="http://www.atmel.com/dyn/resources/prod_documents/doc6310.pdf">
00063  *     GNU-Based Software Development</a> application note or to the
00064  *     <a href="ftp://ftp.iar.se/WWWfiles/arm/Guides/EWARM_UserGuide.ENU.pdf">
00065  *     IAR EWARM User Guide</a>, depending on your chosen solution.
00066  *  -# On the computer, open and configure a terminal application
00067  *     (e.g. HyperTerminal on Microsoft Windows) with these settings:
00068  *    - 115200 bauds
00069  *    - 8 bits of data
00070  *    - No parity
00071  *    - 1 stop bit
00072  *    - No flow control
00073  *  -# Start the application.
00074  *  -# In the terminal window, the following text should appear:
00075  *  \code
00076  *  -- USB Device Audio LoopREC Example xxx --
00077  *  -- AT91xxxxxx-xx
00078  *  -- Compiled: xxx xx xxxx xx:xx:xx --
00079  *  \endcode
00080  *  -# When connecting USB cable to windows, the LED blinks, and the host
00081  *     reports a new USB device attachment (if it's the first time you connect
00082  *     an audio speaker demo board to your host). You can find new
00083  *     "USB Composite Device" and "USB Audio Device" appear in the hardware
00084  *     device list.
00085  *  -# You can play sound in host side through the USB Audio Device.
00086  *     When playing sound, you can also record through the USB Audio Device on
00087  *     the host.
00088  *
00089  *  \section References
00090  *  - usb_audio_looprec/main.c

00097  */

00241 /**
00242  *  Invoked when a frame has been received.
00243  */
00244 static void FrameReceived(uint32_t unused,
00245                           uint8_t status,
00246                           uint32_t transferred,
00247                           uint32_t remaining)
00248 {
00249     if (status == USBD_STATUS_SUCCESS)
00250     {
00251         /* Loopback! add this buffer to write list */
00252         if (!isRecActive)  {}
00253         else
00254         {
00255             AUDDSpeakerPhoneDriver_Write(buffers[inBufferIndex],
00256                                          AUDDevice_BYTESPERFRAME);
00257         }
00258 
00259         /* Update input status data */
00260         bufferSizes[inBufferIndex] = transferred
00261                                         / AUDDevice_BYTESPERSAMPLE;
00262         inBufferIndex = (inBufferIndex + 1) % BUFFER_NUMBER;
00263         numBuffersToSend++;
00264 
00265     }
00266     else if (status == USBD_STATUS_ABORTED)
00267     {
00268         /* Error , ABORT, add NULL buffer */
00269         bufferSizes[inBufferIndex] = 0;
00270         inBufferIndex = (inBufferIndex + 1) % BUFFER_NUMBER;
00271         numBuffersToSend++;
00272     }
00273     else
00274     {
00275         /* Packet is discarded */
00276     }
00277 
00278     /* Receive next packet */
00279     AUDDSpeakerPhoneDriver_Read(buffers[inBufferIndex],
00280                            AUDDevice_BYTESPERFRAME,
00281                            (TransferCallback) FrameReceived,
00282                            0); // No optional argument
00283 }
