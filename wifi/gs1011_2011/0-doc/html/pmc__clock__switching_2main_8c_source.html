00030 /**
00031  * \page pmc_clock_switching PMC Clock Switching Example
00032  *
00033  * \section Purpose
00034  * This example shows how to switch system clock from one to another (PLLA, 
00035  * UPLL, SLCK, MAINCK) or change divider.
00036  *
00037  * \section Requirements
00038  *
00039  * This package can be used with AT91SAM9xx5 evaluation kits.
00040  *
00041  * \section Description
00042  *
00043  * Upon startup, the program configure PIOs for DBGU, PCK and bottons. The baud 
00044  * rate of DBGU is configured as 1200 bps. The application prints the current 
00045  * configuration (except 32Khz slow clock ) and waits for button pressed or 
00046  * input from PC terminal application to switch the system clock to next 
00047  * configuration. PCK1 Outputs can be selected from the clocks provided by the 
00048  * clock (PLLA, PLLB, SLCK, MAINCK) and driven on the pin PCK1 on PA17 
00049  * (Peripheral B).
00050  * After the clock switchs, the PCK1 output signal can be measured by scope 
00051  * compared with the clock configuration.
00052  *
00053  * <ul>
00054  * <li> The Clock Generator integrates a 32,768 Hz low-power oscillator.
00055  * The user can select the crystal oscillator to be the slow clock source,  
00056  * as it provides a more accurate frequency. The command is made by function 
00057  * PmcSlowClockSwitchXtalOsc().</li>
00058  * <li> The Master Clock is selected from one of the clocks provided by the  
00059  * Clock Generator. 
00060  * Selecting the Slow Clock provides a Slow Clock signal to the whole device. 
00061  * Selecting the Main Clock saves power consumption of the PLLs.
00062  * Function _PmcMasterClockSelection() descripts the deatil how to switch master
00063  * clock between difference source.</li>
00064  * </ul>
00065  * \section Usage
00066  *
00067  * -# Build the program and download it inside the evaluation board. Please
00068  *    refer to the
00069  *    <a href="http://www.atmel.com/dyn/resources/prod_documents/6421B.pdf">
00070  *    SAM-BA User Guide</a>, the
00071  *    <a href="http://www.atmel.com/dyn/resources/prod_documents/doc6310.pdf">
00072  *    GNU-Based Software Development</a> application note or to the
00073  *    <a href="ftp://ftp.iar.se/WWWfiles/arm/Guides/EWARM_UserGuide.ENU.pdf">
00074  *    IAR EWARM User Guide</a>, depending on your chosen solution.
00075  * -# On the computer, open and configure a terminal application
00076  *    (e.g. HyperTerminal on Microsoft Windows) with these settings:
00077  *   - 1200 bauds
00078  *   - 8 bits of data
00079  *   - No parity
00080  *   - 1 stop bit
00081  *   - No flow control
00082  * -# Start the application.
00083  * -# In the terminal window, the following text should appear:
00084  *    \code
00085  *     -- PMC clock Switching Example xxx --
00086  *     -- AT91xxxxxx-xx
00087  *     -- Compiled: xxx xx xxxx xx:xx:xx --
00088  *    \endcode
00089  * -# Press one of the buttons listed in the menu to perform the corresponding 
00090  *    action or type "'" to do the same thing if no button is available.
00091  *
00092  * \section References
00093  * - pmc_clock_switching/main.c
00094  * - pio.h
00095  * - pio_it.h
00096  */

00269 /**
00270  * \brief Selection of Master Clock.
00271  *
00272  * \param css  The Master Clock divider source.
00273  * \param pres Master Clock prescaler.
00274  */
00275 static void _PmcMasterClockSelection( uint32_t css, uint32_t pres )
00276 {
00277     uint32_t timeout = 0;
00278     switch (css) {
00279         case PMC_MCKR_CSS_PLLA_CLK: 
00280         case PMC_MCKR_CSS_UPLL_CLK:
00281             /* Program the PRES field in the PMC_MCKR register. */
00282             REG_PMC_MCKR = (REG_PMC_MCKR & (~PMC_MCKR_PRES_Msk) ) | pres;
00283             /* Wait for the MCKRDY bit to be set in the PMC_SR register. */
00284             for (timeout = 0; !(REG_PMC_SR & PMC_SR_MCKRDY) && (timeout++ < CLOCK_TIMEOUT); );
00285             /* Program the CSS field in the PMC_MCKR register. */
00286             REG_PMC_MCKR = (REG_PMC_MCKR & (~PMC_MCKR_CSS_Msk) ) | css;
00287             /* Wait for the MCKRDY bit to be set in the PMC_SR register. */
00288             for (timeout = 0; !(REG_PMC_SR & PMC_SR_MCKRDY) && (timeout++ < CLOCK_TIMEOUT); );
00289             break;
00290 
00291         case PMC_MCKR_CSS_SLOW_CLK:
00292         case PMC_MCKR_CSS_MAIN_CLK:
00293             /* Program the CSS field in the PMC_MCKR register. */
00294             REG_PMC_MCKR = (REG_PMC_MCKR & (~PMC_MCKR_CSS_Msk) ) | css;
00295             /* Wait for the MCKRDY bit to be set in the PMC_SR register. */
00296             for (timeout = 0; !(REG_PMC_SR & PMC_SR_MCKRDY) && (timeout++ < CLOCK_TIMEOUT); );
00297             /* Program the PRES field in the PMC_MCKR register. */
00298             REG_PMC_MCKR = (REG_PMC_MCKR & (~PMC_MCKR_PRES_Msk) ) | pres;
00299             /* Wait for the MCKRDY bit to be set in the PMC_SR register. */
00300             for (timeout = 0; !(REG_PMC_SR & PMC_SR_MCKRDY) && (timeout++ < CLOCK_TIMEOUT); );
00301             break;
00302 
00303         default:
00304             break;
00305     }
00306 
00307     return;

00311 /**
00312  * \brief Select the crystal oscillator to be the source of the slow clock.
00313  */
00314 static void _PmcSlowClockSwitchXtalOsc( void )
00315 {
00316     REG_SCK_CR;
00317     
00318     /* Enable the 32,768 Hz oscillator by setting the bit OSC32EN to 1.*/
00319     REG_SCK_CR |= 1 << 1;
00320     /* Wait 32,768 Hz Startup Time for clock stabilization (software loop). */
00321     _delay(32768);
00322     /* Switch from internal 32 kHz RC to 32,768 Hz oscillator by setting the bit OSCSEL to 1.*/
00323     REG_SCK_CR |= 1 << 3;
00324     /* Wait 5 slow clock cycles for internal resynchronization. */
00325     _delay(32768);
00326     /* Disable the 32 kHz RC oscillator by setting the bit RCEN to 0.*/
00327     REG_SCK_CR &= 0xE;
00328 }

00342 /**
00343  * \brief Restore working clock.
00344  * Here working clock must be running from PLL
00345  * and external crystal oscillator is used.
00346  */
00347 static void _RestoreWorkingClock( uint32_t MorVal, uint32_t PllarVal, uint32_t MckrVal )
00348 {
00349     uint32_t tmp;
00350 
00351     /* switch to slow clock first */
00352     if ((PMC->PMC_MCKR & PMC_MCKR_CSS_Msk) != PMC_MCKR_CSS_SLOW_CLK)
00353     {
00354         PMC->PMC_MCKR = (PMC->PMC_MCKR & (uint32_t)~PMC_MCKR_CSS_Msk)
00355                       | (PMC_MCKR_CSS_SLOW_CLK) ;
00356         while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) ) ;
00357     }
00358     PMC->PMC_MCKR = (PMC->PMC_MCKR & (uint32_t)~PMC_MCKR_PRES_Msk)
00359                   | (PMC_MCKR_PRES_CLOCK) ;
00360 
00361     /* Restart Main Oscillator */
00362     PMC->CKGR_MOR = CKGR_MOR_KEY(0x37)
00363                   | (MorVal & CKGR_MOR_MOSCXTST_Msk)
00364                   | (CKGR_MOR_MOSCRCEN)
00365                   | (CKGR_MOR_MOSCXTEN) ;
00366     while ( !(PMC->PMC_SR & PMC_SR_MOSCXTS) ) ;
00367     /* Switch to moscsel */
00368     if (MorVal & CKGR_MOR_MOSCSEL)
00369     {
00370         PMC->CKGR_MOR = CKGR_MOR_KEY(0x37)
00371                       | CKGR_MOR_MOSCXTST(0x3F)
00372                       | CKGR_MOR_MOSCRCEN
00373                       | CKGR_MOR_MOSCXTEN
00374                       | CKGR_MOR_MOSCSEL ;
00375         while ( !(PMC->PMC_SR & PMC_SR_MOSCSELS) ) ;
00376     }
00377     /* Restart PLL A */
00378     if ( (PllarVal & CKGR_PLLAR_MULA_Msk) != 0 )
00379     {
00380         PMC->CKGR_PLLAR = PllarVal ;
00381         while ( !(PMC->PMC_SR & PMC_SR_LOCKA) ) ;
00382     }
00383     /* Wait for master clock */
00384     while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) ) ;
00385 
00386     /* Switch to main oscillator */
00387     tmp = PMC->PMC_MCKR ;
00388     tmp &= ~(PMC_MCKR_MDIV_Msk|PMC_MCKR_PLLADIV2) ;
00389     tmp |=  (PMC_MCKR_MDIV_Msk|PMC_MCKR_PLLADIV2) & MckrVal ;
00390     PMC->PMC_MCKR = tmp ;
00391     /* Wait for master clock */
00392     while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) ) ;
00393     
00394     /* Switch to main oscillator + prescaler */
00395     tmp = PMC->PMC_MCKR ;
00396     tmp &= ~PMC_MCKR_PRES_Msk ;
00397     tmp |=  PMC_MCKR_PRES_Msk & MckrVal ;
00398     PMC->PMC_MCKR = tmp ;
00399     /* Wait for master clock */
00400     while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) ) ;
00401 
00402     /* Switch to fast clock */
00403     tmp = PMC->PMC_MCKR ;
00404     tmp &= ~PMC_MCKR_CSS_Msk;
00405     tmp |=  PMC_MCKR_CSS_Msk & MckrVal ;
00406     PMC->PMC_MCKR = tmp ;
00407     while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) ) ;
00408 
00409     /* Done */
00410     PMC->PMC_MCKR = MckrVal ;
00411     while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) ) ;
00412 }
