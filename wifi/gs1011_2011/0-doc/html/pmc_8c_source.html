00042 /**
00043  * \brief Enables the clock of a peripheral. The peripheral ID is used
00044  * to identify which peripheral is targetted.
00045  *
00046  * \note The ID must NOT be shifted (i.e. 1 << ID_xxx).
00047  *
00048  * \param id  Peripheral ID (ID_xxx).
00049  */
00050 extern void PMC_EnablePeripheral( uint32_t dwId )
00051 {
00052     assert( dwId < 32 ) ;
00053 
00054     if ( (PMC->PMC_PCSR & ((uint32_t)1 << dwId)) == ((uint32_t)1 << dwId) )
00055     {
00056         TRACE_DEBUG( "PMC_EnablePeripheral: clock of peripheral"  " %u is already enabled\n\r", dwId ) ;
00057     }
00058     else
00059     {
00060         PMC->PMC_PCER = 1 << dwId ;
00061     }
00062 }
