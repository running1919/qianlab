00073  /**
00074  * \brief Initialize the Periodic Interval Timer to generate a tick at the 
00075  * specified period, given the current master clock frequency.
00076  *
00077  *  \param uperiod  Period in uSecond.
00078  *  \param pit_frequency  Master clock frequency in MHz.
00079  */
00080 
00081 void PIT_Init(uint32_t period, uint32_t pit_frequency)
00082 {
00083     PIT->PIT_MR = period? (period * pit_frequency + 8) >> 4 : 0;
00084     PIT->PIT_MR |= PIT_MR_PITEN;
00085 }

00087 /**
00088  * \brief Set the Periodic Interval Value of the PIT.
00089  *
00090  *  \param piv  PIV value to set.
00091  */
00092 void PIT_SetPIV(uint32_t piv)
00093 {
00094     uint32_t dwMr = PIT->PIT_MR & (~PIT_MR_PIV_Msk);
00095     PIT->PIT_MR = dwMr | PIT_MR_PIV(piv);
00096 }
