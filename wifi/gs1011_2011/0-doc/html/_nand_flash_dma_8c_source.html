00054 /**
00055  * \brief Configure the DMA Channels.
00056  * Channels are disabled after configure.
00057  * \returns 0 if the dma channel configuration successfully; otherwise returns

00059  */

00061 {
00062     uint32_t dwCfg;
00063     /* Configure the allocated DMA channel. */
00064     dwCfg = 0
00065            | DMAC_CFG_SRC_PER( 0 )

00067            | DMAC_CFG_FIFOCFG_HALF_CFG;

00069     /* Allocate a DMA channel for NAND RX. */
00070     nandDmaRxChannel = DMAD_AllocateChannel( pDmad, DMAD_TRANSFER_MEMORY, DMAD_TRANSFER_MEMORY);
00071     {
00072         if ( nandDmaRxChannel == DMAD_ALLOC_FAILED ) 
00073         {
00074             return NandCommon_ERROR_DMA;
00075         }
00076     }
00077     
00078     if (DMAD_PrepareChannel( pDmad, nandDmaRxChannel, dwCfg )) return NandCommon_ERROR_DMA;

00080     dmaNandRxDesc.dwDstAddr = 0;
00081     dmaNandRxDesc.dwCtrlA   = 0;
00082     dmaNandRxDesc.dwCtrlB   = DMAC_CTRLB_SIF_AHB_IF1 | DMAC_CTRLB_DIF_AHB_IF0 
00083                             | DMAC_CTRLB_SRC_DSCR | DMAC_CTRLB_DST_DSCR 
00084                             | DMAC_CTRLB_FC_MEM2MEM_DMA_FC
00085                             | DMAC_CTRLB_SRC_INCR_FIXED | DMAC_CTRLB_DST_INCR_INCREMENTING;
00086     if (DMAD_PrepareSingleTransfer( pDmad, nandDmaRxChannel, &dmaNandRxDesc )) return NandCommon_ERROR_DMA;
00087 
00088 
00089     /* Allocate a DMA channel for NAND TX. */

00098     dmaNandTxDesc.dwSrcAddr = 0;

00110 /**
00111  * \brief Configure the DMA Channels for Tx.
00112  * \param ramAddr Source address to be transfered.
00113  * \param size Transfer size in byte.
00114  * \returns 0 if the dma channel configuration and transfer successfully; otherwise returns
00115  * NandCommon_ERROR_XXX.
00116  */
00117 uint8_t NandFlashDmaTransferRam2Nand( uint32_t ramAddr, uint32_t size )
00118 { 
00119     if (!pCurrentDma) return NandCommon_ERROR_DMA;
00120     dmaNandTxDesc.dwSrcAddr = ramAddr;
00121     dmaNandTxDesc.dwDstAddr = (uint32_t) BOARD_NF_DATA_ADDR;
00122     dmaNandTxDesc.dwCtrlA   = ((size + 3) >> 2) | DMAC_CTRLA_SRC_WIDTH_WORD | DMAC_CTRLA_DST_WIDTH_WORD;
00123     if (DMAD_PrepareSingleTransfer( pCurrentDma, nandDmaTxChannel, &dmaNandTxDesc )) return NandCommon_ERROR_DMA;
00124     if (DMAD_StartTransfer( pCurrentDma, nandDmaTxChannel )) 
00125         return NandCommon_ERROR_DMA;
00126     while (DMAD_IsTransferDone( pCurrentDma, nandDmaTxChannel ));
00127     return 0;
00128 }
