00052 /**
00053  * \brief Reads the data and/or spare of a page of a nandflash chip, and verify that
00054  * the data is valid using the ECC information contained in the spare. If one
00055  * buffer pointer is 0, the corresponding area is not saved.
00056  * \param ecc  Pointer to an EccNandFlash instance.
00057  * \param block  Number of block to read from.
00058  * \param page  Number of page to read inside given block.
00059  * \param data  Data area buffer.
00060  * \param spare  Spare area buffer.
00061  * \return 0 if the data has been read and is valid; otherwise returns either
00062  * NandCommon_ERROR_CORRUPTEDDATA or ...
00063  */
00064 static uint8_t EccNandFlash_ReadPageWithSoftwareEcc(
00065     const struct EccNandFlash *ecc,
00066     uint16_t block,
00067     uint16_t page,
00068     void *data,
00069     void *spare)
00070 {
00071     uint8_t tmpSpare[NandCommon_MAXPAGESPARESIZE];
00072     uint8_t error;
00073 
00074     uint8_t tmpData[NandCommon_MAXPAGEDATASIZE];
00075     uint8_t hamming[NandCommon_MAXSPAREECCBYTES];
00076 
00077     uint16_t pageDataSize = NandFlashModel_GetPageDataSize(MODEL(ecc));
00078     uint8_t pageSpareSize = NandFlashModel_GetPageSpareSize(MODEL(ecc));
00079 
00080     /* Start by reading the spare data */
00081     error = RawNandFlash_ReadPage(RAW(ecc), block, page, 0, tmpSpare);
00082     if (error) {
00083 
00084         TRACE_ERROR("EccNandFlash_ReadPage: Failed to read page\n\r");
00085         return error;
00086     }
00087 
00088     /* Then reading the data */
00089     error = RawNandFlash_ReadPage(RAW(ecc), block, page, tmpData, 0);
00090     if (error) {
00091 
00092         TRACE_ERROR("EccNandFlash_ReadPage: Failed to read page\n\r");
00093         return error;
00094     }
00095 
00096     /* Retrieve ECC information from page and verify the data */
00097     NandSpareScheme_ReadEcc(NandFlashModel_GetScheme(MODEL(ecc)), tmpSpare, hamming);
00098     error = Hamming_Verify256x(tmpData, pageDataSize, hamming);
00099     if (error && (error != Hamming_ERROR_SINGLEBIT)) {
00100 
00101         TRACE_ERROR("EccNandFlash_ReadPage: at B%d.P%d Unrecoverable data\n\r",
00102                     block, page);
00103         return NandCommon_ERROR_CORRUPTEDDATA;
00104     }
00105     /* Copy data and/or spare into final buffers */
00106     if (data) {
00107 
00108         memcpy(data, tmpData, pageDataSize);
00109     }
00110     if (spare) {
00111 
00112         memcpy(spare, tmpSpare, pageSpareSize);
00113     }
00114     return 0;
00115 }

00193 /**
00194  * \brief Writes the data and/or spare area of a nandflash page, after calculating an
00195  * ECC for the data area and storing it in the spare. If no data buffer is
00196  * provided, the ECC is read from the existing page spare. If no spare buffer
00197  * is provided, the spare area is still written with the ECC information
00198  * calculated on the data buffer.
00199  * \param ecc Pointer to an EccNandFlash instance.
00200  * \param block  Number of the block to write in.
00201  * \param page  Number of the page to write inside the given block.
00202  * \param data  Data area buffer, can be 0.
00203  * \param spare  Spare area buffer, can be 0.
00204  * \return 0 if successful; otherwise returns an error code.
00205  */
00206 static uint8_t EccNandFlash_WritePageWithSoftwareEcc(
00207     const struct EccNandFlash *ecc,
00208     uint16_t block,
00209     uint16_t page,
00210     void *data,
00211     void *spare)
00212 {
00213     uint8_t error;
00214     uint8_t tmpSpare[NandCommon_MAXPAGESPARESIZE];
00215     uint8_t hamming[NandCommon_MAXSPAREECCBYTES];
00216     uint16_t pageDataSize = NandFlashModel_GetPageDataSize(MODEL(ecc));
00217     uint16_t pageSpareSize = NandFlashModel_GetPageSpareSize(MODEL(ecc));
00218 
00219     /* Compute ECC on the new data, if provided */
00220     /* If not provided, hamming code set to 0xFFFF.. to keep existing bytes */
00221     memset(hamming, 0xFF, NandCommon_MAXSPAREECCBYTES);
00222     if ( data ) {
00223 
00224         /* Compute hamming code on data */
00225         Hamming_Compute256x(data, pageDataSize, hamming);
00226     }
00227 
00228     /* Store code in spare buffer (if no buffer provided, use a temp. one) */
00229     if ( !spare ) {
00230         spare = tmpSpare;
00231         memset(spare, 0xFF, pageSpareSize);
00232     }
00233     NandSpareScheme_WriteEcc(NandFlashModel_GetScheme(MODEL(ecc)), spare, hamming);
00234 
00235     /* Perform write operation */
00236     error = RawNandFlash_WritePage(RAW(ecc), block, page, data, spare);
00237     if ( error ) {
00238         TRACE_ERROR("EccNandFlash_WritePage: Failed to write page\n\r");
00239         return error;
00240     }
00241     return 0;
00242 }
