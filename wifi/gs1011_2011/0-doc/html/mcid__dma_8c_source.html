00227     sSdmmcCommand *pCmd = pMcid->pCmd;

00235     if (pMcid->dwXfrNdx >= totalSize)
00236     {
00237         return 0;

00243     if (bRd)
00244     {

00247     /* M to P: Max size is M size */

00250         maxXSize = bMByte ? MAX_DMA_SIZE : (MAX_DMA_SIZE * 4);

00253     pMcid->dwXSize = totalSize - pMcid->dwXfrNdx;

00256         pMcid->dwXSize = maxXSize;
00257     }
00258     /* Prepare DMA transfer */
00259     if (bRd)
00260     {
00261         /* RX: bytes or words */

00263                                                   toWCOUNT(pMcid->dwXSize));
00264         td.dwSrcAddr = (uint32_t)&(pHw->HSMCI_RDR);
00265         td.dwDstAddr = memAddress;

00270                                  : DMAC_CTRLA_DST_WIDTH_WORD)

00275                        | DMAC_CTRLB_DST_INCR_INCREMENTING
00276                        | DMAC_CTRLB_IEN
00277                        ;
00278         td.dwDscAddr = 0;
00279     }
00280     else
00281     {
00282         /* TX: Always words */
00283         uint32_t btW = DMAC_CTRLA_BTSIZE(toWCOUNT(pMcid->dwXSize));
00284         td.dwSrcAddr = memAddress;
00285         td.dwDstAddr = (uint32_t)&(pHw->HSMCI_TDR);

00287                        | (bMByte ? DMAC_CTRLA_SRC_WIDTH_BYTE

00289                        | (bFByte ? DMAC_CTRLA_DST_WIDTH_BYTE
00290                                  : DMAC_CTRLA_DST_WIDTH_WORD)

00295                        | DMAC_CTRLB_DST_INCR_FIXED

00325 /**
00326  * Configure the  MCI CLKDIV in the MCI_MR register. The max. for MCI clock is
00327  * MCK/2 and corresponds to CLKDIV = 0
00328  * \param pMci  Pointer to the low level MCI driver.
00329  * \param mciSpeed  MCI clock speed in Hz, 0 will not change current speed.
00330  * \param mck       MCK to generate MCI Clock, in Hz
00331  * \return The actual speed used, 0 for fail.
00332  */
00333 static uint32_t MCI_SetSpeed( sMcid* pMci, uint32_t mciSpeed, uint32_t mck )
00334 {
00335     Hsmci *pMciHw = pMci->pMciHw;
00336     uint32_t clkdiv;
00337 
00338     assert(pMci);
00339     assert(pMciHw);
00340 
00341     /* Multimedia Card Interface clock (MCCK or MCI_CK) is Master Clock (MCK)
00342      * divided by (2*(CLKDIV+1))
00343      * mciSpeed = MCK / (2*(CLKDIV+1)) */
00344     if (mciSpeed > 0) {
00345         clkdiv = (mck / 2 / mciSpeed);
00346         /* Speed should not bigger than expired one */
00347         if (mciSpeed < mck/2/clkdiv)
00348         {
00349             clkdiv ++;
00350         }
00351         if ( clkdiv > 0 )
00352         {
00353             clkdiv -= 1;
00354         }
00355         assert( (clkdiv & 0xFFFFFF00) == 0 ) ; /* "mciSpeed too small" */
00356     }
00357     else
00358     {
00359         clkdiv = 0 ;
00360     }
00361     /* Actual MCI speed */
00362     mciSpeed = mck / 2 / (clkdiv + 1);
00363 
00364     /* Modify MR */
00365     HSMCI_DivCtrl( pMciHw, clkdiv, HSMCI_MR_CLKDIV(0x7) );
00366     return (mciSpeed);
00367 }
