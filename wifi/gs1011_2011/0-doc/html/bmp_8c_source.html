00159 /**
00160  * \brief Loads a BMP image located at the given address, decodes it and stores the
00161  * resulting image inside the provided buffer. Image must have the specified
00162  * width & height.
00163  * If no buffer is provided, this function simply checks if it is able to
00164  * decode the image.
00165  * \param file  Buffer which holds the BMP file.
00166  * \param buffer  Buffer in which to store the decoded image.
00167  * \param width  Buffer width in pixels.
00168  * \param height  Buffer height in pixels.
00169  * \param bpp  Number of bits per pixels that the buffer stores.
00170  * \return 0 if the image has been loaded; otherwise returns an error code.
00171  */
00172 uint8_t BMP_Decode( void *file, uint8_t *buffer, uint32_t width, uint32_t height, uint8_t bpp )
00173 {
00174     BMPHeader *header;
00175     uint32_t i, j;
00176     uint8_t r, g, b;
00177     uint8_t *image;
00178 
00179     // Read header information
00180     header = (BMPHeader*) file;
00181 
00182     // Verify that the file is valid
00183     if ( !BMP_IsValid( file ) )
00184     {
00185         TRACE_ERROR("BMP_Decode: File type is not 'BM' (0x%04X).\n\r",header->type);
00186 
00187         return 1;
00188     }
00189 
00190     // Check that parameters match
00191     if ( (header->compression != 0) || (header->width != width) || (header->height != height))
00192     {
00193         TRACE_ERROR("BMP_Decode: File format not supported\n\r");
00194         TRACE_ERROR(" -> .compression = %u\n\r", header->compression);
00195         TRACE_ERROR(" -> .width = %u\n\r", header->width);
00196         TRACE_ERROR(" -> .height = %u\n\r", header->height);
00197         TRACE_ERROR(" -> .bits = %d\n\r", header->bits);
00198 
00199         return 2;
00200     }
00201 
00202     // Get image data
00203     image = (uint8_t *) ((uint32_t) file + header->offset);
00204 
00205     // Check that the bpp resolution is supported
00206     // Only a 24-bit output & 24- or 8-bit input are supported
00207     if ( bpp != 24 )
00208     {
00209         TRACE_ERROR("BMP_Decode: Output resolution not supported\n\r");
00210 
00211         return 3;
00212     }
00213     else
00214     {
00215         if (header->bits == 24)
00216         {
00217             // Decoding is ok
00218             if (!buffer) return 0;
00219 
00220             // Get image data (swapping red & blue)
00221             for ( i=0 ; i < height ; i++ )
00222             {
00223                 for ( j=0 ; j < width; j++ )
00224                 {
00225                     r = image[((height - i - 1) * width + j) * 3 + 2];
00226                     g = image[((height - i - 1) * width + j) * 3 + 1];
00227                     b = image[((height - i - 1) * width + j) * 3];
00228 
00229     #if defined(BOARD_LCD_RGB565)
00230                     // Interlacing
00231                     r = ((r << 1) & 0xF0) | ((g & 0x80) >> 4) | ((r & 0x80) >> 5);
00232                     g = (g << 1) & 0xF8;
00233                     b = b & 0xF8;
00234 
00235                     buffer[(i * width + j) * 3] = b;
00236                     buffer[(i * width + j) * 3 + 1] = g;
00237                     buffer[(i * width + j) * 3 + 2] = r;
00238 
00239     #else
00240                     buffer[(i * width + j) * 3] = r;
00241                     buffer[(i * width + j) * 3 + 1] = g;
00242                     buffer[(i * width + j) * 3 + 2] = b;
00243     #endif //#if defined(BOARD_LCD_RGB565)
00244                 }
00245             }
00246         }
00247         else
00248         {
00249             if ( header->bits == 8 )
00250             {
00251                 // Decoding is ok
00252                 if (!buffer) return 0;
00253 
00254                 // Retrieve palette
00255                 BMPPaletteEntry palette[256];
00256                 memcpy( palette, (uint8_t *) ((uint32_t) file + sizeof( BMPHeader )), header->offset - sizeof( BMPHeader ) ) ;
00257 
00258                 // Decode image (reversing row order)
00259                 for ( i=0 ; i < height ; i++ )
00260                 {
00261                     for (j=0; j < width; j++)
00262                     {
00263                         r = palette[image[(height - i - 1) * width + j]].r;
00264                         g = palette[image[(height - i - 1) * width + j]].g;
00265                         b = palette[image[(height - i - 1) * width + j]].b;
00266 
00267                         buffer[(i * width + j) * 3] = r;
00268                         buffer[(i * width + j) * 3 + 1] = g;
00269                         buffer[(i * width + j) * 3 + 2] = b;
00270                     }
00271                 }
00272             }
00273             else
00274             {
00275 
00276                 TRACE_ERROR("BMP_Decode: Input resolution not supported\n\r");
00277                 TRACE_INFO("header->bits 0x%X \n\r", header->bits);
00278                 return 4 ;
00279             }
00280         }
00281     }
00282 
00283     return 0 ;
00284 }

00286 /**
00287  * \brief Convert RGB 565 to RGB 555 (RGB 555 is adapted to LCD)
00288  *
00289  * \param fileSource  Buffer which holds the RGB file
00290  * \param fileDestination  Buffer in which to store the decoded image
00291  * \param width  Buffer width in pixels.
00292  * \param height  Buffer height in pixels.
00293  * \param bpp  Number of bits per pixels that the buffer stores.
00294   */
00295 void RGB565toBGR555( uint8_t *fileSource, uint8_t *fileDestination, uint32_t width, uint32_t height, uint8_t bpp )
00296 {
00297     uint32_t i;
00298     uint32_t j;
00299     uint32_t row;
00300 
00301     for (i=0; i < height*(bpp/8); i++)
00302     {
00303         row = (i*width*(bpp/8));
00304 
00305         for (j=0; j <= width*(bpp/8); j+=2)
00306         {
00307             fileDestination[row+j] = ((fileSource[row+j+1]>>3)&0x1F)
00308                                     | (fileSource[row+j]&0xE0);
00309             fileDestination[row+j+1] = (fileSource[row+j+1]&0x03)
00310                                     | ((fileSource[row+j]&0x1F)<<2);
00311         }
00312     }
