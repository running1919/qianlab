00034  * A AT25 instance has to be initialized using the Dataflash levle function
00035  * AT25_Configure(). AT25 Dataflash can be automatically detected using

00044  * <li> Writes data on the At25 at the specified address using AT25D_Write().</li>

00341 /**
00342  * \brief Writes data at the specified address on the serial firmware dataflash. The
00343  * page(s) to program must have been erased prior to writing. This function
00344  * handles page boundary crossing automatically.
00345  *
00346  * \param pAt25  Pointer to an AT25 driver instance.
00347  * \param pData  Data buffer.

00352  * been an error during the data programming.
00353  */

00355     At25 *pAt25,
00356     unsigned char *pData,
00357     unsigned int size,
00358     unsigned int address)
00359 {

00361     unsigned int writeSize;
00362     unsigned char error;
00363     unsigned char status;
00364     unsigned int i = 0;
00365     
00366     assert(pAt25);

00369     /* Retrieve device page size */
00370     pageSize = AT25_PageSize(pAt25);
00371 
00372     /* Program one page after the other */
00373     while (size > 0) {
00374         /* Compute number of bytes to program in page */
00375         writeSize = min(size, pageSize - (address % pageSize));
00376 
00377         /* Enable critical write operation */
00378         AT25D_EnableWrite(pAt25);
00379 
00380         /* Program page */

00390             /* Poll the Serial flash status register until the operation is achieved */

00394                 error = AT25_SendCommand(pAt25, AT25_SEQUENTIAL_PROGRAM_1, 1,
00395                                    pData + i, 2, 0, 0, 0);
00396 
00397                 assert(!error);

00399                 /* Wait for transfer to finish */
00400                 AT25D_Wait(pAt25);
00401                 /* Poll the Serial flash status register until the operation is achieved */
00402                 AT25D_WaitReady(pAt25);
00403             }
00404         
00405         }

00410             assert(!error);
