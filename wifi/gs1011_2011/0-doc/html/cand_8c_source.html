00264 /**
00265  * Handler for messages
00266  * \param pCand Pointer to CAN Driver instance.
00267  */
00268 static void CAND_MessageHandler(sCand *pCand)
00269 {
00270     Can *pCan = pCand->pHw;
00271     sCandTransfer *pXfr;
00272     uint8_t bMb;
00273     uint32_t dwMsr;
00274     for (bMb = 0; bMb < CAN_NUM_MAILBOX; bMb ++)
00275     {
00276         /* Mailbox used ? */
00277         pXfr = pCand->pMbs[bMb];
00278         if (pXfr == NULL)
00279             continue;
00280         /* Mailbox ready ? */
00281         dwMsr = CAN_GetMessageStatus(pCan, bMb);
00282         if ((dwMsr & CAN_MSR_MRDY) != CAN_MSR_MRDY)
00283             continue;
00284         /* Handle data */
00285         switch (CAN_GetMessageMode(pCan, bMb) & CAN_MMR_MOT_Msk)
00286         {
00287             case CAN_MMR_MOT_MB_RX_OVERWRITE: /** Next data overwrite current */
00288                 /*pXfr->bState = CAND_XFR_RX_ONE;*/
00289             case CAN_MMR_MOT_MB_RX:
00290             case CAN_MMR_MOT_MB_CONSUMER:   /** TX then RX message */
00291                 pXfr->bMsgLen = (dwMsr & CAN_MSR_MDLC_Msk) >> CAN_MSR_MDLC_Pos;
00292                 CAN_GetMessage(pCan, bMb, pXfr->msgData);
00293                 CAND_EndXfr(pCand, pXfr, CAND_OK);
00294                 break;
00295 
00296             case CAN_MMR_MOT_MB_TX:
00297             case CAN_MMR_MOT_MB_PRODUCER:   /** RX then TX message */
00298                 CAND_EndXfr(pCand, pXfr, CAND_OK);
00299                 break;
00300 
00301             default:
00302                 TRACE_ERROR("MB[%d] disabled\n\r", bMb);
00303                 CAND_EndXfr(pCand, pXfr, CAND_ERROR);
00304                 break;
00305         }
00306         /*if (pXfr->bState != CAND_XFR_RX_ONE)*/
00307         {
00308             /* Disable mailbox interrupt */
00309             CAN_DisableIt(pCan, 1 << bMb);
00310             /* Unlink transfer */
00311             pCand->pMbs[bMb] = NULL;
00312         }
00313     }
00314     /* All transfer finished ? */
00315     if ((CAN_GetItMask(pCan)&CAN_MB_EVENTS) == 0)
00316         pCand->bState = CAND_STATE_ACTIVATED;
00317 }

00533 /**

00535  * \note For data receiving, if there is previous pending message in

00538  * \param bmMbs Mailbox bitmap.

00541 {
00542     Can *pCan = pCand->pHw;
00543     sCandTransfer *pTx;
00544     uint8_t bMb;
00545     uint32_t bmTx = 0;
00546     uint32_t  bmRx = 0;
00547     uint32_t dwMMR;
00548     /* Scan mailboxes that not started */
00549     for (bMb = 0; bMb < CAN_NUM_MAILBOX; bMb ++)
00550     {
00551         if ((bmMbs & (1 << bMb)) == 0)
00552             continue;
00553         /* Check if the mailbox is ready to transfer */
00554         pTx = pCand->pMbs[bMb];
00555         if (pTx == NULL)
00556         {
00557             /* Ignore the mailbox */
00558             bmMbs &= ~(1 << bMb);
00559             continue;
00560         }
00561         if (pTx->bState > CAND_XFR_IDLE)
00562         {
00563             /* Ignore the mailbox */
00564             bmMbs &= ~(1 << bMb);
00565             continue;
00566         }
00567         dwMMR = CAN_GetMessageMode(pCan, bMb);
00568         /* Disabled ? */
00569         if ( 0 == dwMMR )
00570         {
00571             /* Ignore the mailbox */
00572             bmMbs &= ~(1 << bMb);
00573             continue;
00574         }
00575         /* RX ? */
00576         else if ((dwMMR & CAN_MMR_MOT_Msk) <= CAN_MMR_MOT_MB_RX_OVERWRITE)
00577         {
00578             bmRx |= 1 << bMb;
00579         }
00580         /* TX ! */
00581         else
00582         {
00583             bmTx |= 1 << bMb;
00584         }
00585 
00586         /* Change transfer state */
00587         pTx->bState = CAND_XFR_TX;
00588 
00589         /* Nothing to start */
00590         if (bmMbs == 0)
00591             return;
00592     }
00593     /* Change CAN state */
00594     pCand->bState = CAND_STATE_XFR;
00595     /* Start transfers */
00596     CAN_Command(pCan, bmTx);
00597     /* Enable interrupts */
