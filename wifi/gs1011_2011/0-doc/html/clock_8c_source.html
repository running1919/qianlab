00084 /**
00085  * \brief Sets the specified clock configuration.
00086  *
00087  * \param configuration  Index of the configuration to set.
00088  */
00089 void CLOCK_SetConfig(uint8_t configuration)
00090 {
00091     TRACE_DEBUG("Setting clock configuration #%d ... ", configuration);
00092     currentConfig = configuration;
00093 
00094     /* Switch to main oscillator in two operations */
00095     //C->PMC_MCKR = (PMC->PMC_MCKR & (uint32_t)~PMC_MCKR_CSS) | PMC_MCKR_CSS_MAIN_CLK;
00096     while ((PMC->PMC_SR & PMC_SR_MCKRDY) == 0);
00097 
00098     /* Configure PLL */
00099     PMC->CKGR_PLLAR = clockConfigurations[configuration].pllr;
00100     while ((PMC->PMC_SR & PMC_SR_LOCKA) == 0);
00101 
00102     /* Configure master clock in two operations */
00103     //C->PMC_MCKR = (clockConfigurations[configuration].mckr & (uint32_t)~PMC_MCKR_CSS) | PMC_MCKR_CSS_MAIN_CLK;
00104     while ((PMC->PMC_SR & PMC_SR_MCKRDY) == 0);
00105     PMC->PMC_MCKR = clockConfigurations[configuration].mckr;
00106     while ((PMC->PMC_SR & PMC_SR_MCKRDY) == 0);
00107 
00108     /* DBGU reconfiguration */
00109     DBGU_Configure(115200, clockConfigurations[configuration].mck*1000000);
00110     TRACE_DEBUG("done.\n\r");
00111 }
