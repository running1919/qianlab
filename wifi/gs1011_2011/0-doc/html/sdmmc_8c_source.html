00353 /**
00354  * Find SDIO ManfID, Fun0 tuple.
00355  * \param pSd         Pointer to \ref sSdCard instance.
00356  * \param address     Search area start address.
00357  * \param size        Search area size.
00358  * \param pAddrManfID Pointer to ManfID address value buffer.
00359  * \param pAddrFunc0  Pointer to Func0 address value buffer.
00360  */
00361 static uint8_t SdioFindTuples(sSdCard *pSd,
00362                               uint32_t address, uint32_t size,
00363                               uint32_t *pAddrManfID,
00364                               uint32_t *pAddrFunc0)
00365 {
00366     uint8_t error, tmp[3];
00367     uint32_t addr = address;
00368     uint8_t  flagFound = 0; /* 1:Manf, 2:Func0 */
00369     uint32_t addManfID = 0, addFunc0 = 0;
00370     for(;flagFound != 3;) {
00371         error = SDIO_ReadDirect(pSd, SDIO_CIA, addr, tmp, 3);
00372         if (error) {
00373             TRACE_ERROR("SdioFindTuples.RdDirect: %d\n\r", error);
00374             return error;
00375         }
00376         /* End */
00377         if (tmp[0] == CISTPL_END) break;
00378         /* ManfID */
00379         else if (tmp[0] == CISTPL_MANFID) {
00380             flagFound |= 1; addManfID = addr;
00381         }
00382         /* Func0 */
00383         else if (tmp[0] == CISTPL_FUNCE && tmp[2] == 0x00) {
00384             flagFound |= 2; addFunc0 = addr;
00385         }
00386         /* Tuple error ? */
00387         if (tmp[1] == 0) break;
00388         /* Next address */
00389         addr += (tmp[1] + 2);
00390         if (addr > (address + size)) break;
00391     }
00392     if (pAddrManfID) *pAddrManfID = addManfID;
00393     if (pAddrFunc0)  *pAddrFunc0  = addFunc0;
00394     return 0;
00395 }

02078 /**
02079  * \brief Run the SD/MMC/SDIO Mode initialization sequence.
02080  * This function runs the initialization procedure and the identification
02081  * process. Then it leaves the card in ready state. The following procedure must
02082  * check the card type and continue to put the card into tran(for memory card)
02083  * or cmd(for io card) state for data exchange.
02084  * \param pSd  Pointer to a SD card driver instance.
02085  * \return 0 if successful; otherwise returns an \ref sdmmc_rc "SD_ERROR code".
02086  */
02087 static uint8_t SdMmcIdentify(sSdCard *pSd)
02088 {
02089     uint8_t mem = 0, io = 0, f8 = 0, mp = 1, ccs = 0;
02090     uint32_t status;
02091     uint8_t error;
02092     /* Reset HC to default HS and BusMode */
02093     _HwSetHsMode(pSd, 0);
02094     _HwSetBusWidth(pSd, 1);
02095     /* Reset SDIO: CMD52, write 1 to RES bit in CCCR (bit 3 of register 6) */
02096     status = SDIO_RES;
02097     error = Cmd52(pSd, 1, SDIO_CIA, 0, SDIO_IOA_REG, &status);
02098     if (!error && ((status & STATUS_SDIO_R5)==0)){}
02099     else if (error == SDMMC_ERROR_NORESPONSE){}
02100     else {
02101         TRACE_DEBUG("SdMmcIdentify.Cmd52 fail: %u, %x\n\r", error, status);
02102     }
02103     /* Reset MEM: CMD0 */
02104     error = SwReset(pSd, 1);
02105     if (error) {
02106         TRACE_DEBUG("SdMmcIdentify.SwReset fail: %u\n\r", error);
02107     }
02108 
02109     /* CMD8 is newly added in the Physical Layer Specification Version 2.00 to
02110      * support multiple voltage ranges and used to check whether the card
02111      * supports supplied voltage. The version 2.00 host shall issue CMD8 and
02112      * verify voltage before card initialization.
02113      * The host that does not support CMD8 shall supply high voltage range... */
02114     error = SdCmd8(pSd, 1);
02115     if (error == 0) f8 = 1;
02116     else if (error != SDMMC_ERROR_NORESPONSE) {
02117         TRACE_ERROR("SdMmcIdentify.Cmd8: %d\n\r", error);
02118         return SDMMC_ERROR;
02119     }
02120     /* Delay after "no response" */
02121     else Delay(8000);
02122 
02123     /* CMD5 is newly added for SDIO initialize & power on */
02124     status = 0;
02125     error = Cmd5(pSd, &status);
02126     if (error) {
02127         TRACE_INFO("SdMmcIdentify.Cmd5: %d\n\r", error)
02128     }
02129     /* Card has SDIO function */
02130     else if ((status & SDIO_OCR_NF) > 0) {
02131         unsigned int cmd5Retries = 10000;
02132         do {
02133             status &= SD_HOST_VOLTAGE_RANGE;
02134             error = Cmd5(pSd, &status);
02135             if (status & SD_OCR_BUSY) break;
02136         } while(!error && cmd5Retries --);
02137         if (error) {
02138             TRACE_ERROR("SdMmcIdentify.Cmd5 V: %d\n\r", error);
02139             return SDMMC_ERROR;
02140         }
02141         io = 1;
02142         TRACE_INFO("SDIO\n\r");
02143         /* IO only ?*/
02144         mp = ((status & SDIO_OCR_MP) > 0);
02145     }
02146     /* Has memory: SD/MMC/COMBO */
02147     if (mp) {
02148         /* Try SD memory initialize */
02149         error = Acmd41(pSd, f8, &ccs);
02150         if (error) {
02151             unsigned int cmd1Retries = 10000;
02152             TRACE_DEBUG("SdMmcIdentify.Acmd41: %u, try MMC\n\r", error);
02153             /* Try MMC initialize */
02154             error = SwReset(pSd, 10);
02155             if (error) {
02156                 TRACE_ERROR("SdMmcIdentify.Mmc.SwReset: %d\n\r", error);
02157                 return SDMMC_ERROR;
02158             }
02159             ccs = 1;
02160             do { error = Cmd1(pSd, &ccs); }while(error && cmd1Retries -- > 0);
02161             if (error) {
02162                 TRACE_ERROR("SdMmcIdentify.Cmd1: %d\n\r", error);
02163                 return SDMMC_ERROR;
02164             }
02165             else if (ccs) pSd->bCardType = CARD_MMCHD;
02166             else          pSd->bCardType = CARD_MMC;
02167             /* MMC card identification OK */
02168             TRACE_INFO("MMC Card\n\r");
02169             return 0;
02170         }
02171         else if (ccs) { TRACE_INFO("SDHC MEM\n\r");}
02172         else          { TRACE_INFO("SD MEM\n\r");}
02173         mem = 1;
02174     }
02175     /* SD(IO) + MEM ? */
02176     if (!mem) {
02177         if (io) pSd->bCardType = CARD_SDIO;
02178         else {
02179             TRACE_ERROR("Unknown card\n\r");
02180             return SDMMC_ERROR;
02181         }
02182     }
02183     /* SD(HC) combo */
02184     else if (io)
02185         pSd->bCardType = ccs ? CARD_SDHCCOMBO : CARD_SDCOMBO;
02186     /* SD(HC) */
02187     else
02188         pSd->bCardType = ccs ? CARD_SDHC : CARD_SD;
02189 
02190     return 0;

02195  * This function runs after the initialization and identification procedure. It
02196  * gets all necessary information from the card and deside transfer block size,
02197  * clock speed and bus width. It sets the SD/MMC/SDIO card in transfer
02198  * (or command) state.
02199  * \param pSd  Pointer to a SD card driver instance.
02200  * \return 0 if successful; otherwise returns an \ref sdmmc_rc "SD_ERROR code".
02201  */
02202 static uint8_t SdMmcEnum(sSdCard *pSd)
02203 {
02204     uint8_t mem , io;
02205     uint8_t error;
02206     uint32_t ioSpeed = 0, memSpeed = 0;
02207     uint8_t hsExec = 0, bwExec = 0;
02208 
02209     /* - has Memory/IO/High-Capacigy - */
02210     mem = ((pSd->bCardType & CARD_TYPE_bmSDMMC) > 0);
02211     io  = ((pSd->bCardType & CARD_TYPE_bmSDIO)  > 0);
02212 
02213     /* For MEMORY cards:
02214      * The host then issues the command ALL_SEND_CID (CMD2) to the card to get
02215      * its unique card identification (CID) number.
02216      * Card that is unidentified (i.e. which is in Ready State) sends its CID
02217      * number as the response (on the CMD line). */
02218     if (mem) {
02219         error = Cmd2(pSd);
02220         if (error) {
02221             TRACE_ERROR("SdMmcInit.cmd2(%d)\n\r", error);
02222             return error;
02223         }
02224     }
02225

02227      * Thereafter, the host issues CMD3 (SEND_RELATIVE_ADDR) asks the
02228      * card to publish a new relative card address (RCA), which is shorter than
02229      * CID and which is used to address the card in the future data transfer
02230      * mode. Once the RCA is received the card state changes to the Stand-by
02231      * State. At this point, if the host wants to assign another RCA number, it
02232      * can ask the card to publish a new number by sending another CMD3 command
02233      * to the card. The last published RCA is the actual RCA number of the
02234      * card. */
02235     error = Cmd3(pSd);
02236     if (error) {
02237         TRACE_ERROR("SdMmcInit.cmd3(%d)\n\r", error);
02238         return error;
02239     }
02240 
02241     /* For MEMORY cards:
02242      * SEND_CSD (CMD9) to obtain the Card Specific Data (CSD register),
02243      * e.g. block length, card storage capacity, etc... */
02244     if (mem) {
02245         error = Cmd9(pSd);
02246         if (error) {
02247             TRACE_ERROR("SdMmcInit.cmd9(%d)\n\r", error);
02248             return error;
02249         }
02250     }
02251 
02252     /* Now select the card, to TRAN state */
02253     error = MmcSelectCard(pSd, CARD_ADDR(pSd), 0);
02254     if (error) {
02255         TRACE_ERROR("SdMmcInit.SelCard(%d)\n\r", error);
02256         return error;
02257     }
02258 
02259     /* - Now in TRAN, obtain extended setup information - */
02260 
02261     /* If the card support EXT_CSD, read it! */
02262     TRACE_INFO("Card Type %d, CSD_STRUCTURE %u\n\r",
02263                pSd->bCardType, SD_CSD_STRUCTURE(pSd));
02264 
02265     /* Get extended information of the card */
02266     SdMmcUpdateInformation(pSd, 0, 1);
02267 
02268     /* Calculate transfer speed */
02269     if (io)     ioSpeed = SdioGetMaxSpeed(pSd);
02270     if (mem)    memSpeed = SdmmcGetMaxSpeed(pSd);
02271     /* Combo, min speed */
02272     if (io && mem) {
02273         pSd->dwTranSpeed = (ioSpeed > memSpeed) ? memSpeed : ioSpeed;
02274     }
02275     /* SDIO only */
02276     else if (io) {
02277         pSd->dwTranSpeed = ioSpeed;
02278     }
02279     /* Memory card only */
02280     else if (mem) {
02281         pSd->dwTranSpeed = memSpeed;
02282     }
02283     pSd->dwTranSpeed *= 1000;
02284 
02285     /* Enable more bus width Mode */
02286     error = SdMmcDesideBuswidth(pSd);
02287     if (!error) bwExec = 1;
02288     else if (error != SDMMC_ERROR_NOT_SUPPORT) {
02289         TRACE_ERROR("SdmmcEnum.DesideBusWidth: %d\n\r", error);
02290         return SDMMC_ERROR;
02291     }
02292 
02293     /* Enable High-Speed Mode */
02294     error = SdMmcEnableHighSpeed(pSd);
02295     if (!error) hsExec = 1;
02296     else if (error != SDMMC_ERROR_NOT_SUPPORT) {
02297         TRACE_ERROR("SdmmcEnum.EnableHS: %d\n\r", error);
02298         return SDMMC_ERROR;
02299     }
02300 
02301     /* In HS mode transfer speed *2 */
02302     if (hsExec) pSd->dwTranSpeed *= 2;
02303 
02304     /* Update card information since status changed */
02305     if (bwExec || hsExec) SdMmcUpdateInformation(pSd, hsExec, 1);
02306 
02307     return 0;
02308 }

02545 /**
02546  * Run the SDcard initialization sequence. This function runs the
02547  * initialisation procedure and the identification process, then it sets the
02548  * SD card in transfer state to set the block length and the bus width.
02549  * \return 0 if successful; otherwise returns an \ref sdmmc_rc "error code".
02550  * \param pSd  Pointer to a SD card driver instance.
02551  */
02552 uint8_t SD_Init(sSdCard *pSd)
02553 {
02554     uint8_t  error;
02555     uint32_t clock;
02556 
02557     _SdParamReset(pSd);
02558 
02559     /* Set low speed for device identification (LS device max speed) */
02560     clock = 400000;
02561     _HwSetClock(pSd, &clock);
02562 
02563     /* Initialization delay: The maximum of 1 msec, 74 clock cycles and supply
02564      * ramp up time. Supply ramp up time provides the time that the power is
02565      * built up to the operating level (the bus master supply voltage) and the
02566      * time to wait until the SD card can accept the first command. */
02567     /* Power On Init Special Command */
02568     error = Pon(pSd);
02569     if (error) {
02570         TRACE_ERROR("SD_Init.PowON:%d\n\r", error);
02571         return error;
02572     }
02573 
02574     /* After power-on or CMD0, all cards?
02575      * CMD lines are in input mode, waiting for start bit of the next command.
02576      * The cards are initialized with a default relative card address
02577      * (RCA=0x0000) and with a default driver stage register setting
02578      * (lowest speed, highest driving current capability). */
02579     error = SdMmcIdentify(pSd);
02580     if (error) {
02581         TRACE_ERROR("SD_Init.Identify: %d\n\r", error);
02582         return error;
02583     }
02584     error = SdMmcEnum(pSd);
02585     if (error) {
02586         TRACE_ERROR("SD_Init.Enum: %d\n\r", error);
02587         return error;
02588     }
02589 
02590     /* In the case of a Standard Capacity SD Memory Card, this command sets the
02591      * block length (in bytes) for all following block commands
02592      * (read, write, lock).
02593      * Default block length is fixed to 512 Bytes.
02594      * Set length is valid for memory access commands only if partial block read
02595      * operation are allowed in CSD.
02596      * In the case of a High Capacity SD Memory Card, block length set by CMD16
02597      * command does not affect the memory read and write commands. Always 512
02598      * Bytes fixed block length is used. This command is effective for
02599      * LOCK_UNLOCK command.
02600      * In both cases, if block length is set larger than 512Bytes, the card sets
02601      * the BLOCK_LEN_ERROR bit. */
02602     if (pSd->bCardType == CARD_SD) {
02603         error = Cmd16(pSd, SDMMC_BLOCK_SIZE);
02604     }
02605     pSd->wCurrBlockLen = SDMMC_BLOCK_SIZE;
02606 
02607     /* Reset status for R/W */
02608     pSd->bState = SDMMC_STATE_TRAN;
02609 
02610     /* If MMC Card & get size from EXT_CSD */
02611     if ((pSd->bCardType & CARD_TYPE_bmSDMMC) == CARD_TYPE_bmMMC
02612         && SD_CSD_C_SIZE(pSd->CSD) == 0xFFF) {
02613         pSd->dwNbBlocks = MMC_EXT_SEC_COUNT(pSd->EXT);
02614         /* Block number less than 0x100000000/512 */
02615         if (pSd->dwNbBlocks > 0x800000)
02616             pSd->dwTotalSize = 0xFFFFFFFF;
02617         else
02618             pSd->dwTotalSize = MMC_EXT_SEC_COUNT(pSd->EXT)*512;
02619     }
02620     /* If SD CSD v2.0 */
02621     else if((pSd->bCardType & CARD_TYPE_bmSDMMC) == CARD_TYPE_bmSD
02622         && SD_CSD_STRUCTURE(pSd->CSD) >= 1) {
02623         pSd->wBlockSize  = 512;
02624         pSd->dwNbBlocks  = SD_CSD_BLOCKNR_HC(pSd->CSD);
02625         pSd->dwTotalSize = 0xFFFFFFFF;
02626     }
02627     /* Normal SD/MMC card */
02628     else if (pSd->bCardType & CARD_TYPE_bmSDMMC) {
02629         pSd->wBlockSize  = SD_CSD_BLOCK_LEN(pSd->CSD);
02630         pSd->dwTotalSize = SD_CSD_TOTAL_SIZE(pSd->CSD);
02631         pSd->dwNbBlocks  = SD_CSD_BLOCKNR(pSd->CSD);
02632     }
02633 
02634     if (pSd->bCardType == CARD_UNKNOWN) {
02635         return SDMMC_ERROR_NOT_INITIALIZED;
02636     }
02637     /* Automatically select the max clock */
02638     clock = pSd->dwTranSpeed;
02639     _HwSetClock(pSd, &clock);
02640     TRACE_WARNING_WP("-I- Set SD/MMC clock to %uK\n\r", clock/1000);
02641     pSd->dwCurrSpeed = clock;
02642     return 0;
02643 }
