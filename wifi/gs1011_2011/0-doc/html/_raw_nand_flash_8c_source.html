00116  static void WriteColumnAddress(
00117     const struct RawNandFlash *raw,
00118     uint16_t columnAddress)
00119 {
00120     uint16_t pageDataSize = NandFlashModel_GetPageDataSize(MODEL(raw));
00121 
00122     /* Check the data bus width of the NandFlash */
00123     if (NandFlashModel_GetDataBusWidth(MODEL(raw)) == 16) {
00124         /* Div 2 is because we address in word and not in byte */
00125         columnAddress >>= 1;
00126     }
00127     /* Send single column address byte for small block devices, or two column address bytes for large block devices*/
00128     while (pageDataSize > 2) {
00129 
00130         if (NandFlashModel_GetDataBusWidth(MODEL(raw)) == 16) {
00131             WRITE_ADDRESS16(raw, columnAddress & 0xFF);
00132         }
00133         else {
00134             WRITE_ADDRESS(raw, columnAddress & 0xFF);
00135         }
00136         pageDataSize >>= 8;
00137         columnAddress >>= 8;
00138     }
00139 }

00334 /**
00335  * \brief Reads the data and/or the spare areas of a page of a NandFlash into the
00336  * provided buffers. If a buffer pointer is 0, the corresponding area is not
00337  * read.
00338  * \param raw  Pointer to a RawNandFlash instance.
00339  * \param block  Number of the block where the page to read resides.
00340  * \param page  Number of the page to read inside the given block.
00341  * \param data  Buffer where the data area will be stored.
00342  * \param spare  Buffer where the spare area will be stored.
00343  * \return 0 if the operation has been successful; otherwise returns 1.
00344  */
00345 static uint8_t ReadPage(
00346     const struct RawNandFlash *raw,
00347     uint16_t block,
00348     uint16_t page,
00349     void *data,
00350     void *spare)
00351 {
00352     uint8_t hasSmallBlocks = NandFlashModel_HasSmallBlocks(MODEL(raw));
00353     uint32_t pageDataSize = NandFlashModel_GetPageDataSize(MODEL(raw));
00354     uint32_t pageSpareSize = NandFlashModel_GetPageSpareSize(MODEL(raw));
00355     uint32_t colAddress;
00356     uint32_t rowAddress;
00357     uint8_t eccStatus;
00358 
00359     assert ( data || spare); /* RawNandFlash_ReadPage: At least one area must be read */
00360     
00361     TRACE_DEBUG("RdPg(B#%d:P#%d)\r\n", block, page);
00362 
00363     /* Calculate actual address of the page */
00364     rowAddress = block * NandFlashModel_GetBlockSizeInPages(MODEL(raw)) + page;
00365 
00366     /* Start operation */
00367     ENABLE_CE( raw );
00368 
00369     if ( data ) {
00370         colAddress = 0;
00371     }
00372     else {
00373         /* to read spare area in sequential access */
00374         colAddress = pageDataSize;
00375     }
00376 
00377     /* Use either small blocks or large blocks data area read */
00378     if ( hasSmallBlocks ) {
00379         if( colAddress == 0) {
00380             WRITE_COMMAND( raw, COMMAND_READ_A);
00381         }
00382         else if(colAddress == pageDataSize) {
00383             WRITE_COMMAND( raw, COMMAND_READ_C);
00384         }
00385         WriteColumnAddress( raw, colAddress);
00386         WriteRowAddress( raw, rowAddress);
00387     }
00388     else {
00389 
00390         WRITE_COMMAND( raw, COMMAND_READ_1);
00391         WriteColumnAddress( raw, colAddress);
00392         WriteRowAddress( raw, rowAddress);
00393         WRITE_COMMAND( raw, COMMAND_READ_2);
00394     }
00395 
00396     /* Wait for the nand to be ready */
00397     WaitReady( raw );
00398 
00399     /* Read data area if needed */
00400     if ( data ) {
00401         WRITE_COMMAND(raw, COMMAND_READ_1);
00402         ReadData(raw, (uint8_t *) data, pageDataSize);
00403 
00404         if ( spare ) {
00405             ReadData( raw, (uint8_t *) spare, pageSpareSize);
00406         }
00407     }
00408     else {
00409         /* Read spare area only */
00410         WRITE_COMMAND( raw, COMMAND_READ_1);
00411         ReadData( raw, (uint8_t *) spare, pageSpareSize);
00412     }
00413     if (NandGetEccStatus() == ECC_INTERNAL)
00414     {
00415         /* When internal ECC is enabled, the READ STATUS (70h) command is required after the
00416            completion of the data transfer to determine whether an uncorrectable read
00417            error occured. */
00418         WRITE_COMMAND(raw, COMMAND_STATUS);
00419         eccStatus = (READ_DATA8(raw) & STATUS_ERROR);
00420         DISABLE_CE( raw );
00421         return eccStatus;
00422     }
00423     /* Disable CE */
00424     DISABLE_CE( raw );
00425     return 0;
00426 }

00428 /**
00429  * \brief Reads the data and/or the spare areas of a page of a NandFlash into the
00430  * provided buffers. If a buffer pointer is 0, the corresponding area is not
00431  * read.
00432  * \param raw  Pointer to a RawNandFlash instance.
00433  * \param block  Number of the block where the page to read resides.
00434  * \param page  Number of the page to read inside the given block.
00435  * \param data  Buffer where the data area will be stored.
00436  * \return 0 if the operation has been successful; otherwise returns 1.
00437  */
00438 static uint8_t ReadPageWithPmecc(
00439     const struct RawNandFlash *raw,
00440     uint16_t block,
00441     uint16_t page,
00442     void *data)
00443 {
00444     uint32_t pageDataSize = NandFlashModel_GetPageDataSize( MODEL(raw) );
00445     uint32_t spareDataSize = NandFlashModel_GetPageSpareSize(MODEL(raw));
00446     uint32_t colAddress;
00447     uint32_t rowAddress;
00448     
00449     TRACE_DEBUG("RdPg(B#%d:P#%d)\r\n", block, page);
00450 
00451     /* Calculate actual address of the page */
00452     rowAddress = block * NandFlashModel_GetBlockSizeInPages( MODEL(raw)) + page;
00453 
00454     PMECC->PMECC_CTRL = PMECC_CTRL_RST;
00455     PMECC->PMECC_CTRL = PMECC_CTRL_ENABLE;
00456     if ((PMECC->PMECC_CFG & PMECC_CFG_SPAREEN) != PMECC_CFG_SPAREEN) {
00457         PMECC->PMECC_CFG |= PMECC_CFG_AUTO;
00458     }
00459     /* Start operation */
00460     ENABLE_CE( raw );
00461     if ( data ) {
00462         colAddress = 0;
00463     }
00464     else {
00465         /* to read spare area in sequential access */
00466         colAddress = pageDataSize;
00467     }
00468     WRITE_COMMAND( raw, COMMAND_READ_1 );
00469     WriteColumnAddress( raw, colAddress );
00470     WriteRowAddress( raw, rowAddress );
00471     WRITE_COMMAND( raw, COMMAND_READ_2 );
00472 
00473     /* Wait for the nand to be ready */
00474     WaitReady( raw );
00475 
00476     WRITE_COMMAND( raw, COMMAND_READ_1 );
00477     /* Reset the ECC module*/
00478     PMECC->PMECC_CTRL |= PMECC_CTRL_RST;
00479     /* Start a Data Phase */
00480     PMECC->PMECC_CTRL |= PMECC_CTRL_DATA;
00481 
00482     /* Spare is written here as well since it is more efficient */
00483     if (PMECC->PMECC_CFG & PMECC_CFG_SPAREEN) 
00484     {
00485         /* When the page has been fully retrieved from NAND, the ECC area is read using the user mode by writing one to the
00486         USER field of the PMECC_CTRL register*/
00487         PMECC->PMECC_CTRL |= PMECC_CTRL_USER;
00488         ReadData( raw, (uint8_t *) data, pageDataSize + spareDataSize );
00489     }
00490     else
00491     {
00492         ReadData( raw, (uint8_t *) data, pageDataSize + (PMECC->PMECC_EADDR + 1) );
00493     }
00494     /* Wait until the kernel of the PMECC is not busy */
00495     while(( PMECC->PMECC_SR) & PMECC_SR_BUSY);
00496     /* Disable CE */
00497     DISABLE_CE( raw );
00498     return 0;
00499 }
