00029 /**
00030  * \page spi_slave SPI Slave Example
00031  *
00032  * \section Purpose
00033  *
00034  * This example uses Serial Peripheral Interface (SPI1) in slave mode to
00035  * communicate to another interface (SPI0) in master mode .
00036  *
00037  * \section Requirements
00038  *
00039  * This package can be used with AT91SAM9xx5 evaluation kits by switching off 
00040  * the dataflash.
00041  * Please connect the SPI pins :
00042  *        - <b>SPI0(MASTER)--SPI1(SLAVE)</b>
00043  *        - SPI0_NPCS0(PA14)--SPI1_NPCS0(PA8)
00044  *        - SPI1_MISO(PA11)--SPI1_MISO(PA21)
00045  *        - SPI1_MOSI(PA12)--SPI1_MOSI(PA22)
00046  *        - SPI0_SPCK(PA13)--SPI1_SPCK(PA23)
00047  *
00048  * \section Descriptions
00049  *
00050  * This example shows control of the SPI slave, how to configure and
00051  * transfer data with SPI slave. By default, example runs in SPI slave mode,
00052  * waiting for SPI slave & DBGU input.
00053  *
00054  * The code can be roughly broken down as follows:
00055  * <ul>
00056  * <li> 't' will start SPI transfer test
00057  * <ol>
00058  * <li>Configure SPI as master, setup SPI clock.
00059  * <li>Send CMD_TEST to indicate the start of test.
00060  * <li>Send several 64-byte blocks, on transmitting next block, the content of last block is returned.
00061  * <li>Send CMD_STATUS command and waiting for the status reprots from slave.
00062  * <li>Send CMD_END command to indicate the end of test.
00063  * </ol>
00064  * <li>Setup SPI clock for master.
00065  * </ul>
00066  *
00067  * \section Usage
00068  *
00069  * -# Compile the application and connect the DBGU port of the evaluation board 
00070  *    to the computer.
00071  * -# Open and configure a terminal application on PC
00072  *    (e.g. HyperTerminal on Microsoft Windows) with these settings:
00073  *   - 115200 bauds
00074  *   - 8 bits of data
00075  *   - No parity
00076  *   - 1 stop bit
00077  *   - No flow control
00078  * -# Download the program inside the evaluation board and run it. Please refer to
00079  *    <a href="http://www.atmel.com/dyn/resources/prod_documents/6421B.pdf">
00080  *    SAM-BA User Guide</a>, the
00081  *    <a href="http://www.atmel.com/dyn/resources/prod_documents/doc6310.pdf">
00082  *    GNU-Based Software Development</a> application note or to the
00083  *    <a href="ftp://ftp.iar.se/WWWfiles/arm/Guides/EWARM_UserGuide.ENU.pdf">
00084  *    IAR EWARM User Guide</a>, depending on your chosen solution.
00085  * -# Upon startup, the application will output the following line on the DBGU:
00086  *    \code
00087  *     -- Spi slave Example  xxx --
00088  *     -- AT91xxxxxx-xx
00089  *     -- Compiled: xxx xx xxxx xx:xx:xx --
00090  *    \endcode
00091  * -# The following traces detail operations on the spl slave example, displaying success
00092  *    or error messages depending on the results of the commands.

00268 static void SpiSlaveCommandProcess( void )
00269 {
00270     if ( spi1Cmd == CMD_END )
00271     {
00272         spi1State = SLAVE_STATE_IDLE;
00273         spi1Status.totalNumBlocks   = 0;
00274         spi1Status.totalNumCommands = 0;
00275     }
00276     else
00277     {
00278         switch( spi1State )
00279         {
00280             case SLAVE_STATE_IDLE:
00281                 /* Only CMD_TEST accepted*/
00282                 if ( spi1Cmd == CMD_TEST )
00283                 {
00284                     spi1State = SLAVE_STATE_TEST ;
00285                 }
00286             break ;
00287 
00288             case SLAVE_STATE_TEST:
00289                 /* Only CMD_DATA accepted*/
00290                 if ( (spi1Cmd & 0xffff0000) == CMD_DATA )
00291                 {
00292                     spi1State = SLAVE_STATE_DATA ;
00293                 }
00294                 numTestBlocks = spi1Cmd & 0x0000ffff ;
00295             break;
00296 
00297             case SLAVE_STATE_DATA:
00298                 spi1Status.totalNumBlocks++ ;
00299 
00300                 if ( spi1Status.totalNumBlocks == numTestBlocks )
00301                 {
00302                     spi1State = SLAVE_STATE_STATUS_ENTRY;
00303                 }
00304             break ;
00305 
00306             case SLAVE_STATE_STATUS_ENTRY :
00307                 spi1State = SLAVE_STATE_STATUS ;
00308             break ;
00309 
00310             case SLAVE_STATE_END:
00311             break;
00312         }
00313     }
00314 }
