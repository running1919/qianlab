00129 /**
00130  * \brief An accurate one-to-one comparison is necessary between PSRAM and SMC waveforms for
00131     a complete SMC configuration.
00132    \note The system is running at 133 MHz for the EBI Bus.
00133          Please refer to the "AC Characteristics" section of the customer product datasheet.
00134  */
00135 static void SMC_ConfigurePsram(void)
00136 {
00137     /* 16 bits MT45W1MW16PDGA Psram for example */
00138 
00139     /* The SMC is clocked through the Power Management Controller (PMC), thus the programmer
00140        must first configure the PMC to enable the SMC clock. */
00141 
00142     printf("-I- Configure PMC to enable the SMC clock.\n\r");
00143     printf("-I- Configure SMC timing and mode,\n\r");
00144     /* NWE_SETUP:     The setup of address and data before the NWE falling edge in write operation.
00145                       (tAS) Address Valid to Write Enable Low min. 0ns.
00146        NCS_WR_SETUP:  The setup time of address before the NCS falling edge in write operation.
00147                       (tAS) Address Valid to Chip Enable Low min. 0ns.
00148        NRD_SETUP:     The setup of address before the NRD falling edge in read operation.
00149        NCS_RD_SETUP:  The setup time of address before the NCS falling edge in read operation. */
00150 
00151     REG_SMC_SETUP0 = 0
00152                     | SMC_SETUP_NWE_SETUP(0)
00153                     | SMC_SETUP_NCS_WR_SETUP(0)
00154                     | SMC_SETUP_NRD_SETUP(0)
00155                     | SMC_SETUP_NCS_RD_SETUP(0);
00156 
00157     /* NWE_CYCLE:     The total duration of the write cycle.
00158                       NWE_CYCLE = NWE_SETUP + NWE_PULSE + NWE_HOLD
00159                       = NCS_WR_SETUP + NCS_WR_PULSE + NCS_WR_HOLD
00160                       (tWC) Write Cycle Time min. 70ns
00161        NRD_CYCLE:     The total duration of the read cycle.
00162                       NRD_CYCLE = NRD_SETUP + NRD_PULSE + NRD_HOLD
00163                       = NCS_RD_SETUP + NCS_RD_PULSE + NCS_RD_HOLD
00164                       (tRC) Read Cycle Time min. 70ns. */
00165     REG_SMC_CYCLE0 = 0
00166                     | SMC_CYCLE_NWE_CYCLE(10)
00167                     | SMC_CYCLE_NRD_CYCLE(10);
00168 
00169 
00170     /* NWE_HOLD:      The hold time of address and data after the NWE rising edge.
00171                       (tWR) Write Enable High to Address Transition min 15ns.
00172        NCS_WR_HOLD:   The hold time of address after the NCS rising edge.
00173                       (tWR)  Write Enable High to Address Transition min 15ns.
00174        NRD_HOLD:      The hold time of address and data after the NRD rising edge.
00175                       (tOHAH) Output Enable High to Address Invalid min. 5ns.
00176        NCS_RD_HOLD:   The hold time of address after the NCS rising edge.
00177                       (tCHAH) Chip Enable High to Address Invalid min. 5ns. */
00178 
00179 
00180     /* NWE_PULSE:     The time between NWE falling edge and NWE rising edge.
00181                       NWE_PULSE  = NWE_CYCLE - NWE_SETUP - NWE_HOLD = tWP = 46ns
00182        NCS_WR_PULSE:  The time between NCS falling edge and NCS rising edge in write operation.
00183                       NCS_WR_PULSE = NWE_CYCLE - NCS_WR_SETUP - NCS_WR_HOLD = 70 - 0 - 15 = 55 ns.
00184        NRD_PULSE:     The time between NRD falling edge and NRD rising edge in read operation.
00185                       NRD_PULSE =  NRD_CYCLE - NRD SETUP - NRD_HOLD = 70 - 10 - 0 = 60 ns.
00186        NCS_RD_PULSE:  The hold time of address after the NCS rising edge.
00187                       NCS_RD_PULSE  = NRD_CYCLE - NCS_RD_SETUP - NCS_RD_HOLD = 70 - 5 - 5 = 60ns. */
00188     REG_SMC_PULSE0 = 0
00189                     | SMC_PULSE_NWE_PULSE(7)
00190                     | SMC_PULSE_NCS_WR_PULSE(7)
00191                     | SMC_PULSE_NRD_PULSE(8)
00192                     | SMC_PULSE_NCS_RD_PULSE(8);
00193 
00194     REG_SMC_MODE0 = SMC_MODE_READ_MODE | SMC_MODE_WRITE_MODE | SMC_MODE_DBW(1);
00195 
00196     printf("-I- SMC Setup Register 0x%08x \n\r", (unsigned int)REG_SMC_SETUP0);
00197     printf("-I- SMC Pulse Register 0x%08x \n\r", (unsigned int)REG_SMC_PULSE0);
00198     printf("-I- SMC Cycle Register 0x%08x \n\r", (unsigned int)REG_SMC_CYCLE0);
00199     printf("-I- SMC MODE Register  0x%08x \n\r", (unsigned int)REG_SMC_MODE0);
00200 }

00202 /**
00203  * \brief Go/No-Go test of the first 10K-Bytes of external PSRAM access.
00204    \return 0 if test is failed else 1.
00205  */
00206 
00207 static uint8_t AccessPsramTest(void)
00208 {
00209     uint32_t i;
00210     uint32_t *ptr = (uint32_t *) PSRAM_BASE_ADDRESS;
00211 
00212     for (i = 0; i < 10 * 1024; ++i) {
00213         if (i & 1) {
00214             ptr[i] = 0x55AA55AA | (1 << i);
00215         }
00216         else {
00217             ptr[i] = 0xAA55AA55 | (1 << i);
00218         }
00219     }
00220     for (i = 0; i < 10 * 1024; ++i) {
00221         if (i & 1) {
00222             if (ptr[i] != (0x55AA55AA | (1 << i))) {
00223                 return 0;
00224             }
00225         }
00226         else {
00227             if (ptr[i] != (0xAA55AA55 | (1 << i))) {
00228                 return 0;
00229             }
00230         }
00231     }
00232     return 1;
00233 }
