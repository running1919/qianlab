00125 /**
00126  * \brief  Get qt1070 key signal value in the given Key. These are the key's
00127  * of 16-bit key signals which are accessed as two 8-bit bytes,stored MSB first
00128  *
00129  * \param pTwid   Pointer to twi driver structure.
00130  * \param key     Key index.
00131  * \return Key signal value.
00132  */
00133 uint16_t QT1070_GetKey_Signal(Twid *pTwid, uint8_t key)
00134 {
00135     uint8_t data[2];
00136     data[0] = QT1070_ReadReg( pTwid, QT1070_REG_KEY0_SIGNAL_MSB + key * 2);
00137     data[1] = QT1070_ReadReg( pTwid, QT1070_REG_KEY0_SIGNAL_LSB + key * 2);
00138     return (data[0] << 8) | data[1];
00139 }
