00379 /**
00380  * Disk test
00381  * \param slot Card slot (not used now).
00382  * \param clr  Do block clear.
00383  * \param wr   Do block write.
00384  * \param rd   Do block read.
00385  */
00386 static void DiskTest(uint8_t iMci,
00387                      uint8_t clr,
00388                      uint8_t wr,
00389                      uint8_t rd)
00390 {
00391     sSdCard *pSd = &sdDrv[iMci];
00392     uint8_t error = 0;
00393     uint32_t i, errcnt = 0;
00394     uint32_t multiBlock, block, splitMulti;
00395 
00396     DumpSeperator();
00397 
00398     if (pSd->bCardType == CARD_SDIO)
00399     {
00400         printf("-!- SDIO only card, please run SDIO example\n\r");
00401         return;
00402     }
00403 
00404     printf("-!- MCI %d, code: 1.clr, 2.wr, 3.rd\n\r", iMci);
00405 
00406     /* Perform tests on each block */
00407     multiBlock = 0;
00408     for (block = TEST_BLOCK_START;
00409          block < TEST_BLOCK_END;
00410          block += multiBlock) {
00411 
00412         /* Perform different single or multiple bloc operations */
00413         if (multiBlock >= 16)   multiBlock <<= 1;
00414         else                    multiBlock ++;
00415         if (multiBlock > NB_MULTI_BLOCKS)
00416             multiBlock = 1;
00417 
00418         /* Multi-block adjustment */
00419         if (block + multiBlock > TEST_BLOCK_END) {
00420             multiBlock = TEST_BLOCK_END - block;
00421         }
00422 
00423         /* ** Perform single block or multi block transfer */
00424         printf("\r-I- Testing block [%6u - %6u] ...", block, (block + multiBlock -1));
00425 
00426         if (clr) {
00427             /* - Clear the block */
00428             memset(pBuffer, 0, SDMMC_BLOCK_SIZE * multiBlock);
00429             for (i=0; i < SDMMC_BLOCK_SIZE * multiBlock; i++) {
00430                 if (pBuffer[i] != 0) {
00431                     /* Fatal error */
00432                     printf("\n\r-F- Data @ %u for write : 0x00 <> 0x%02x\n\r", i, pBuffer[i]);
00433                     return;
00434                 }
00435             }
00436             error = MMCT_WriteFun(pSd, block, multiBlock, pBuffer);
00437             if (error) {
00438                 printf("\n\r-E- 1. Write block (%d) #%u\n\r", error, block);
00439                 if(MaxErrorBreak(0)) return;
00440                 /* Skip following test */
00441                 continue;
00442             }
00443             /* - Read back the data to check the write operation */
00444             memset(pBuffer, 0xFF, SDMMC_BLOCK_SIZE * multiBlock);
00445             error = MMCT_ReadFun(pSd, block, multiBlock, pBuffer);
00446             if (error) {
00447                 printf("\n\r-E- 1. Read block (%d) #%u\n\r", error, block);
00448                 if(MaxErrorBreak(0)) return;
00449                 /* Skip following test */
00450                 continue;
00451             }
00452             for (i=0; i < SDMMC_BLOCK_SIZE * multiBlock; i++) {
00453                 if (pBuffer[i] != 0) {
00454                     printf("\n\r-E- 1. B%u.D[%u] : 0 <> 0x%02X\n\r", (unsigned int)block, i, (int)pBuffer[i]);
00455                     if(MaxErrorBreak(0)) return;
00456                     /* Only find first verify error. */
00457                     break;
00458                 }
00459             }
00460         }
00461 
00462         if (wr) {
00463             /* - Write a checkerboard pattern on the block */
00464             for (i=0; i < SDMMC_BLOCK_SIZE * multiBlock; i++) {
00465                 if ((i & 1) == 0)  pBuffer[i] = (i & 0x55);
00466                 else               pBuffer[i] = (i & 0xAA);
00467             }
00468             for (i = 0; i < multiBlock; ) {
00469                 splitMulti = ((multiBlock - i) > NB_SPLIT_MULTI) ?
00470                                         NB_SPLIT_MULTI : (multiBlock - i);
00471                 error = MMCT_WriteFun(pSd,
00472                                       block + i,
00473                                       splitMulti,
00474                                       &pBuffer[i * SDMMC_BLOCK_SIZE]);
00475                 if (error) break;
00476                 i += splitMulti;
00477             }
00478             if (error)
00479             {
00480                 printf("\n\r-E- 2. Write block #%u(%u+%u): %d\n\r",  (unsigned int)(block+i), (unsigned int)block, (unsigned int)i, error);
00481                 if(MaxErrorBreak(0)) return;
00482                 /* Skip Following Test */
00483                 continue;
00484             }
00485         }
00486 
00487         if (rd) {
00488             /* - Read back the data to check the write operation */
00489             memset(pBuffer, 0, SDMMC_BLOCK_SIZE * multiBlock);
00490             for (i = 0; i < multiBlock; ) {
00491                 splitMulti = ((multiBlock - i) > NB_SPLIT_MULTI) ?
00492                                         NB_SPLIT_MULTI : (multiBlock - i);
00493                 error = MMCT_ReadFun(pSd,
00494                                      block + i,
00495                                      splitMulti,
00496                                      &pBuffer[i * SDMMC_BLOCK_SIZE]);
00497                 if (error) break;
00498                 i += splitMulti;
00499             }
00500             if (error)
00501             {
00502                 printf("\n\r-E- 2. Read block #%u(%u+%u): %d\n\r", (unsigned int)(block + i), (unsigned int)block, (unsigned int)i, error);
00503                 if(MaxErrorBreak(0)) return;
00504                 /* Skip Following Test */
00505                 continue;
00506             }
00507             errcnt = 0;
00508             for (i=0; i < SDMMC_BLOCK_SIZE * multiBlock; i++) {
00509 
00510                 if (!(((i & 1) == 0) && (pBuffer[i] == (i & 0x55))) &&
00511                     !(((i & 1) != 0) && (pBuffer[i] == (i & 0xAA))) ) {
00512                     uint32_t j, js;
00513                     printf("\n\r-E- 2.%d. Data @ %u (0x%x)\n\r", (int)errcnt, (unsigned int)i, (unsigned int)i);
00514                     printf("  -Src:");
00515                     js = (i > 8) ? (i - 8) : 0;
00516                     for (j = js; j < i + 8; j ++)
00517                         printf(" %02x", (unsigned int)(((j & 1)!= 0) ? (j & 0xAA):(j & 0x55)));
00518                     printf("\n\r  -Dat:");
00519                     for (j = js; j < i + 8; j ++)
00520                         printf("%c%02x", (i == j) ? '!' : ' ', pBuffer[j]);
00521                     printf("\n\r");
00522                     if(MaxErrorBreak(0)) return;
00523                     // Only find first 3 verify error.
00524                     if (errcnt ++ >= 3)
00525                         break;
00526                 }
00527             }
00528         }
00529 
00530         if (DBGU_IsRxReady()) {
00531             switch(DBGU_GetChar()) {
00532                 /* Skip 100M */
00533                 case 'k':

00536                         block -= 5 + multiBlock;
00537                     }
00538                     printf("\n\r");
00539                     break;
00540                 /* Cancel */
00541                 case 'c':
00542                     return;
00543             }
00544         }
