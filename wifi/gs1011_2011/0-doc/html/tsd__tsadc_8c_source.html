00100 /**
00101  * Interrupt handler for the TouchScreen.
00102  * Handles pen press, pen move and pen release events
00103  * by invoking three callback functions.
00104  */
00105 void TSD_Handler(uint32_t dwAdcStatus)
00106 {
00107     Adc *pAdc = ADC;
00108 
00109     uint32_t status;
00110 
00111     /* TSADC status */
00112     status  = dwAdcStatus;
00113     status &= /*ADC_GetItMask(pAdc) &*/ TS_STATUSES;
00114     if (status == 0)    return;
00115 
00116     /* Pen released */
00117     if (status & ADC_ISR_NOPEN)
00118     {
00119         if ((bTsFlags & TS_PEN_STAT) == 0)
00120         {
00121             /* Register last data */
00122             memcpy(dwLastTsData, dwTsData, sizeof(dwTsData));
00123             /* Invoke PenReleased callback */
00124             if (TSDCom_IsCalibrationOk())
00125                 TSD_PenReleased(dwTsData[0], dwTsData[1]);
00126         }
00127         bTsFlags = 0;
00128         /* Stop periodic trigger & enable pen */
00129         ADC_SetTsAverage(pAdc, ADC_TSMR_TSAV_NO_FILTER);
00130         ADC_SetTsDebounce(pAdc, BOARD_TOUCHSCREEN_DEBOUNCE);
00131         ADC_SetTriggerMode(pAdc, ADC_TRGR_TRGMOD_PEN_TRIG);
00132         /* Disable pen release detect */
00133         ADC_DisableIt(pAdc, ADC_IDR_NOPEN);
00134         /* Enable pen press detect */
00135         ADC_EnableIt(pAdc, ADC_IER_PEN);
00136     }
00137     /* Pen pressed */
00138     else if (status & ADC_ISR_PEN)
00139     {
00140         bTsFlags |= TS_PEN_STAT;
00141         /* Configure for peripdic trigger */
00142         ADC_SetTsAverage(pAdc, ADC_TSMR_TSAV_AVG8CONV);
00143         ADC_SetTsDebounce(pAdc, 300);         /* 300ns */
00144         ADC_SetTriggerMode(pAdc, ADC_TRGR_TRGMOD_PERIOD_TRIG);
00145         /* Disable pen press detect */
00146         ADC_DisableIt(pAdc, ADC_IDR_PEN);
00147         /* Enable pen release detect */
00148         ADC_EnableIt(pAdc, ADC_IER_NOPEN|ADC_IER_XRDY|ADC_IER_YRDY|ADC_IER_PRDY);
00149     }
00150     else if (status & ADC_ISR_PENS)
00151     {
00152         /* X */
00153         if (status & ADC_ISR_XRDY)
00154         {
00155             bTsFlags |= TS_X_RDY;
00156         }
00157         /* Y */
00158         if (status & ADC_ISR_YRDY)
00159         {
00160             bTsFlags |= TS_Y_RDY;
00161         }
00162         /* P: (X/1024)*[(Z2/Z1)-1] */
00163         if (status & ADC_ISR_PRDY)
00164         {
00165             bTsFlags |= TS_P_RDY;
00166         }
00167     }
00168     /* X,Y,P are ready */
00169     if ((bTsFlags & TS_DATA_RDY) == TS_DATA_RDY)
00170     {
00171         uint32_t xpos, z2, z1;
00172         bTsFlags &= ~TS_DATA_RDY;
00173 
00174         /* Get X,Y */
00175         TSD_GetRawMeasurement(dwRaw);
00176 
00177         /* Interprate X,Y */
00178         TSDCom_InterpolateMeasurement(dwRaw, dwTsData);
00179 
00180         /* Get P: Rp = Rxp*(Xpos/1024)*[(Z2/Z1)-1] */
00181         dwRaw[2] = ADC_GetTsPressure(pAdc);
00182         #ifdef TS_XY_SWAP
00183         xpos = (dwRaw[1]);
00184         #else
00185         xpos = (dwRaw[0]);
00186         #endif
00187         xpos = (xpos & ADC_XPOSR_XPOS_Msk) >> ADC_XPOSR_XPOS_Pos;
00188         z2 = (dwRaw[2] & ADC_PRESSR_Z2_Msk) >> ADC_PRESSR_Z2_Pos;
00189         z1 = (dwRaw[2] & ADC_PRESSR_Z1_Msk) >> ADC_PRESSR_Z1_Pos;
00190         dwTsData[2] = (xpos) * (z2 - z1) / z1;
00191 
00192         /* PenPress */
00193         if (bTsFlags & TS_PEN_STAT)
00194         {
00195             bTsFlags &= ~TS_PEN_STAT;
00196             /* Invoke PenPress callback */
00197             if (TSDCom_IsCalibrationOk())
00198                 TSD_PenPressed(dwTsData[0], dwTsData[1], dwTsData[2]);
00199         }
00200         /* Periodic if data change invoke callback */
00201         if (dwTsData[0] != dwLastTsData[0]
00202             ||   dwTsData[1] != dwLastTsData[1]
00203             ||   dwTsData[2] != dwLastTsData[2] )
00204         {
00205             /* Register last data */
00206             memcpy(dwLastTsData, dwTsData, sizeof(dwTsData));
00207             /* Invoke PenMoved callback */
00208             if (TSDCom_IsCalibrationOk())
00209                 TSD_PenMoved(dwTsData[0], dwTsData[1], dwTsData[2]);
00210         }
00211         
00212     }
00213 }

00219 /**
00220  * Reads and store a touchscreen measurement in the provided array.
00221  * The value stored are:
00222  *  - data[0] = XPOS * 1024 / XSCALE
00223  *  - data[1] = YPOS * 1024 / YSCALE
00224  * \param pData  Array where the measurements will be stored
00225  */
00226 void TSD_GetRawMeasurement(uint32_t *pData)
00227 {
00228     Adc *pAdc = ADC;
00229     uint32_t xr, yr;
00230     #ifdef TS_XY_SWAP
00231     yr = ADC_GetTsXPosition(pAdc);
00232     xr = ADC_GetTsYPosition(pAdc);
00233     #else
00234     xr = ADC_GetTsXPosition(pAdc);
00235     yr = ADC_GetTsYPosition(pAdc);
00236     #endif
00237     pData[0]  = ((xr & ADC_XPOSR_XPOS_Msk) >> ADC_XPOSR_XPOS_Pos) * 1024;
00238     pData[0] /= ((xr & ADC_XPOSR_XSCALE_Msk) >> ADC_XPOSR_XSCALE_Pos);
00239     pData[1]  = ((yr & ADC_YPOSR_YPOS_Msk) >> ADC_YPOSR_YPOS_Pos) * 1024;
00240     pData[1] /= ((yr & ADC_YPOSR_YSCALE_Msk) >> ADC_YPOSR_YSCALE_Pos);
00241 }

00243 /**
00244  * Wait pen pressed
00245  */
00246 void TSD_WaitPenPressed(void)
00247 {
00248     Adc *pAdc = ADC;
00249     uint8_t bFlags = 0;
00250     uint32_t dwStatus;
00251     /* Wait for touch & end of conversion */
00252     while (1)
00253     {
00254         dwStatus = ADC_GetItStatus(pAdc);
00255         if (dwStatus & ADC_ISR_PEN) bFlags |= 8;
00256         if (dwStatus & ADC_ISR_XRDY)bFlags |= 1;
00257         if (dwStatus & ADC_ISR_YRDY)bFlags |= 2;
00258         if (dwStatus & ADC_ISR_PRDY)bFlags |= 4;
00259         if (bFlags == 0xF) break;
00260     }
00261 }
