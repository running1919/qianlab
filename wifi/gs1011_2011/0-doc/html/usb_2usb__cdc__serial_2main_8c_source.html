00029 /** \cond usb_cdc_serial
00030  * \page usb_cdc_serial USB CDC Serial Converter Example
00031  *
00032  * \section Purpose
00033  *
00034  * The USB CDC Serial Project will help you to get familiar with the
00035  * USB Device Port(UDP) and USART interface on AT91SAM microcontrollers. Also
00036  * it can help you to be familiar with the USB Framework that is used for
00037  * rapid development of USB-compliant class drivers such as USB Communication
00038  * Device class (CDC).
00039  *
00040  * \section Requirements
00041  *
00042  * This package can be used with SAM9XX5 evaluation kits.
00043  *
00044  * \section Description
00045  *
00046  * This demo simulates a USB to RS-232 Serial Port Converter.
00047  *
00048  * When an EK running this program connected to a host (PC for example), with
00049  * USB cable, the EK appears as a Seriao COM port for the host, after driver
00050  * installation with the offered 6119.inf. Then the host can send or receive
00051  * data through the port with host software. The data stream from the host is
00052  * then sent to the EK, and forward to USART port of AT91SAM chips. The USART
00053  * port of the EK is monitored by the timer and the incoming data will be sent
00054  * to the host.
00055  *
00056  * \section Usage
00057  *
00058  * -# Build the program and download it inside the evaluation board. Please
00059  *    refer to the
00060  *    <a href="http://www.atmel.com/dyn/resources/prod_documents/doc6421.pdf">
00061  *    SAM-BA User Guide</a>, the
00062  *    <a href="http://www.atmel.com/dyn/resources/prod_documents/doc6310.pdf">
00063  *    GNU-Based Software Development</a> application note or to the
00064  *    <a href="ftp://ftp.iar.se/WWWfiles/arm/Guides/EWARM_UserGuide.ENU.pdf">
00065  *    IAR EWARM User Guide</a>, depending on your chosen solution.
00066  * -# On the computer, open and configure a terminal application
00067  *    (e.g. HyperTerminal on Microsoft Windows) with these settings:
00068  *   - 115200 bauds
00069  *   - 8 bits of data
00070  *   - No parity
00071  *   - 1 stop bit
00072  *   - No flow control
00073  * -# Start the application.
00074  * -# In the terminal window, the following text should appear:
00075  *     \code
00076  *     -- USB Device CDC Serial Project xxx --
00077  *     -- AT91xxxxxx-xx
00078  *     -- Compiled: xxx xx xxxx xx:xx:xx --
00079  *     \endcode
00080  * -# When connecting USB cable to windows, the host
00081  *    reports a new USB device attachment (if it's the first time you connect
00082  *    an %audio speaker demo board to your host). You can use the inf file
00083  *    libraries\\usb\\device\\cdc-serial\\drv\\6119.inf to install the serial
00084  *    port. Then new "AT91 USB to Serial Converter (COMx)" appears in the
00085  *    hardware device list.
00086  * -# You can run hyperterminal to send data to the port. And it can be seen
00087  *    at the other hyperterminal connected to the USART port of the EK.
00088  *
00089  * \section References
00090  * - usb_cdc_serial/main.c
00091  * - usart: USART interface driver
00092  * - tc: TIMER/COUNTER interface driver
00093  * - usb: USB Framework, USB CDC driver and UDP interface driver
00094  *    - \ref usbd_framework
00095  *       - \ref usbd_api

00098  *       - \ref usbd_cdc_host_drv

00304  * USART interrupt handler
00305  */
00306 static void ISR_USART(void)
00307 {
00308     Usart *pUs = BASE_USART;
00309     uint32_t status;
00310     uint16_t serialState;
00311 
00312     status  = USART_GetStatus(pUs);
00313     status &= USART_GetItMask(pUs);
00314 
00315     /* If USB device is not configured, do nothing */
00316     if (!isCdcSerialON)
00317     {
00318         USART_DisableIt(pUs, 0xFFFFFFFF);
00319         return;
00320     }
00321 
00322     /* Errors */
00323     serialState = CDCDSerialDriver_GetSerialState();
00324     /* Overrun */
00325     if ((status & US_CSR_OVRE) != 0)
00326     {
00327         TRACE_WARNING( "USART_IrqHandler: Overrun\n\r");
00328         serialState |= CDCSerialState_OVERRUN;
00329     }
00330     /* Framing error */
00331     if ((status & US_CSR_FRAME) != 0)
00332     {
00333         TRACE_WARNING( "USART_IrqHandler: Framing error\n\r");
00334         serialState |= CDCSerialState_FRAMING;
00335     }
00336     CDCDSerialDriver_SetSerialState(serialState);
00337 }

00341 /**
00342  * Handles interrupts coming from Timer #0.
00343  */
00344 static void ISR_TC0(void)
00345 {
00346     Tc *pTc0 = TC0;
00347     sDmad *pDmad = &dmad;
00348     uint8_t size;
00349     uint32_t dmaChannel = usartDmaRxChannel;
00350     uint8_t  iController = (dmaChannel >> 8);
00351     uint8_t  iChannel    = (dmaChannel) & 0xFF;
00352     uint8_t  nextBuffer = (1 - usartCurrentBuffer);
00353     volatile uint32_t status = pTc0->TC_CHANNEL[0].TC_SR;
00354 
00355     if ((status & TC_SR_CPCS) != 0)
00356     {
00357         tcTick ++;
00358 
00359         if (isCdcSerialON)
00360         {
00361             /* Get Received size */
00362             size = DMAC_GetDestinationAddr(pDmad->pDmacs[iController], iChannel)
00363                     - (uint32_t)(usartBuffers[usartCurrentBuffer]);
00364             if (size == 0)
00365             {
00366                 pTc0->TC_CHANNEL[0].TC_CCR = TC_CCR_CLKEN | TC_CCR_SWTRG;
00367                 return;
00368             }
00369 
00370             /* Stop DMA */
00371             DMAD_StopTransfer(pDmad, dmaChannel);
00372             /* Restart DMA in next buffer */
00373             _UsartDmaRx( nextBuffer );
00374 
00375             /* Debug log: US->USB */
00376             _DebugLog(hdrT2U, sizeof(hdrT2U), usartBuffers[usartCurrentBuffer], size);
00377             /* Send current buffer through the USB */
00378             while (CDCDSerialDriver_Write(usartBuffers[usartCurrentBuffer],
00379                                           size, 0, 0) != USBD_STATUS_SUCCESS);
00380 
00381             usartCurrentBuffer = nextBuffer;
00382         }
00383         pTc0->TC_CHANNEL[0].TC_CCR = TC_CCR_CLKEN | TC_CCR_SWTRG;
00384     }
00385 }
