00030 /**
00031  * \page sysc System Controller Example
00032  *
00033  * \section Purpose
00034  * This example allows to measure the consumption of the core in different modes
00035  * (idle mode, slow clock mode, backup mode) of SAM9XX5.
00036  *
00037  * \section Requirements
00038  *
00039  * This package can be used with AT91SAM9xx5 evaluation kits.
00040  *
00041  * \subsection hw_setup Hardware Setup
00042  *
00043  * - Open jumper JP5 to disable "force power ON" so wakeup function is enabled.
00044  * - Install the battery so backup mode can work.
00045  *
00046  * \section Description
00047  *
00048  * At startup, the program configures all the PIOs as input to avoid
00049  * parasite consumption (except the DBGU PIOs). Then a menu is displayed.
00050  * It allows user to enter in a mode To measure consumption.
00051  *
00052  * An amperemeter has to be plugged on the board instead of the VIN jumper.
00053  *
00054  * \note To observe full function of backup mode, the code should be
00055  *       downloaded to nandflash or serialflash so that the code runs after
00056  *       power wakeup. Please make sure the binary file is less than 24K since
00057  *       it's loaded by ROM boot.
00058  *
00059  * \note Since the GNU generated code exceeds the size limitation of ROM boot,
00060  *       its backup mode can not be tested. You can use EWARM generated binary
00061  *       to test.
00062  *
00063  * \section Usage
00064  *
00065  * -# Build the program and download it inside the evaluation board. Please
00066  *    refer to the
00067  *    <a href="http://www.atmel.com/dyn/resources/prod_documents/6421B.pdf">
00068  *    SAM-BA User Guide</a>, the
00069  *    <a href="http://www.atmel.com/dyn/resources/prod_documents/doc6310.pdf">
00070  *    GNU-Based Software Development</a> application note or to the
00071  *    <a href="ftp://ftp.iar.se/WWWfiles/arm/Guides/EWARM_UserGuide.ENU.pdf">
00072  *    IAR EWARM User Guide</a>, depending on your chosen solution.
00073  * -# On the computer, open and configure a terminal application
00074  *    (e.g. HyperTerminal on Microsoft Windows) with these settings:
00075  *   - 1200 bauds
00076  *   - 8 bits of data
00077  *   - No parity
00078  *   - 1 stop bit
00079  *   - No flow control
00080  * -# Download compiled binary to nandflash or serialflash on EK.
00081  * -# Start the application.
00082  * -# In the terminal window, the following text should appear:
00083  *    \code
00084  *     -- SYSC Example xxx --
00085  *     -- xxxxxx-xx
00086  *     -- Compiled: xxx xx xxxx xx:xx:xx --
00087  *    \endcode

00333 /**
00334  * \brief Restore working clock.
00335  * Here working clock must be running from PLL
00336  * and external crystal oscillator is used.
00337  */
00338 static void _RestoreWorkingClock( uint32_t MorVal, uint32_t PllarVal, uint32_t MckrVal )
00339 {
00340     uint32_t tmp;
00341 
00342     /* switch to slow clock first */
00343     if ((PMC->PMC_MCKR & PMC_MCKR_CSS_Msk) != PMC_MCKR_CSS_SLOW_CLK)
00344     {
00345         PMC->PMC_MCKR = (PMC->PMC_MCKR & (uint32_t)~PMC_MCKR_CSS_Msk)
00346                       | (PMC_MCKR_CSS_SLOW_CLK) ;
00347         while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) ) ;
00348     }
00349     PMC->PMC_MCKR = (PMC->PMC_MCKR & (uint32_t)~PMC_MCKR_PRES_Msk)
00350                   | (PMC_MCKR_PRES_CLOCK) ;
00351 
00352     /* Restart Main Oscillator */
00353     PMC->CKGR_MOR = CKGR_MOR_KEY(0x37)
00354                   | (MorVal & CKGR_MOR_MOSCXTST_Msk)
00355                   | (CKGR_MOR_MOSCRCEN)
00356                   | (CKGR_MOR_MOSCXTEN) ;
00357     while ( !(PMC->PMC_SR & PMC_SR_MOSCXTS) ) ;
00358     /* Switch to moscsel */
00359     if (MorVal & CKGR_MOR_MOSCSEL)
00360     {
00361         PMC->CKGR_MOR = CKGR_MOR_KEY(0x37)
00362                       | CKGR_MOR_MOSCXTST(0x3F)
00363                       | CKGR_MOR_MOSCRCEN
00364                       | CKGR_MOR_MOSCXTEN
00365                       | CKGR_MOR_MOSCSEL ;
00366         while ( !(PMC->PMC_SR & PMC_SR_MOSCSELS) ) ;
00367     }
00368     /* Restart PLL A */
00369     if ( (PllarVal & CKGR_PLLAR_MULA_Msk) != 0 )
00370     {
00371         PMC->CKGR_PLLAR = PllarVal ;
00372         while ( !(PMC->PMC_SR & PMC_SR_LOCKA) ) ;
00373     }
00374     /* Wait for master clock */
00375     while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) ) ;
00376 
00377     /* Switch to main oscillator */
00378     tmp = PMC->PMC_MCKR ;
00379     tmp &= ~(PMC_MCKR_MDIV_Msk|PMC_MCKR_PLLADIV2) ;
00380     tmp |=  (PMC_MCKR_MDIV_Msk|PMC_MCKR_PLLADIV2) & MckrVal ;
00381     PMC->PMC_MCKR = tmp ;
00382     /* Wait for master clock */
00383     while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) ) ;
00384     
00385     /* Switch to main oscillator + prescaler */
00386     tmp = PMC->PMC_MCKR ;
00387     tmp &= ~PMC_MCKR_PRES_Msk ;
00388     tmp |=  PMC_MCKR_PRES_Msk & MckrVal ;
00389     PMC->PMC_MCKR = tmp ;
00390     /* Wait for master clock */
00391     while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) ) ;
00392 
00393     /* Switch to fast clock */
00394     tmp = PMC->PMC_MCKR ;
00395     tmp &= ~PMC_MCKR_CSS_Msk;
00396     tmp |=  PMC_MCKR_CSS_Msk & MckrVal ;
00397     PMC->PMC_MCKR = tmp ;
00398     while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) ) ;
00399 
00400     /* Done */
00401     PMC->PMC_MCKR = MckrVal ;
00402     while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) ) ;
00403 }

00405 /**
00406  * \brief Select the crystal oscillator to be the source of the slow clock.
00407  */
00408 static void _PmcSlowClockSwitchXtalOsc( void )
00409 {
00410 /* SCK register */
00411 #define REG_SCK_CR      (*(volatile uint32_t*)0xFFFFFE50)
00412     REG_SCK_CR;
00413     
00414     /* Enable the 32,768 Hz oscillator by setting the bit OSC32EN to 1.*/
00415     REG_SCK_CR |= 1 << 1;
00416     /* Wait 32,768 Hz Startup Time for clock stabilization (software loop). */
00417     _Delay(32768);
00418     /* Switch from internal 32 kHz RC to 32,768 Hz oscillator by setting the bit OSCSEL to 1.*/
00419     REG_SCK_CR |= 1 << 3;
00420     /* Wait 5 slow clock cycles for internal resynchronization. */
00421     _Delay(32768);
00422     /* Disable the 32 kHz RC oscillator by setting the bit RCEN to 0.*/
00423     REG_SCK_CR &= 0xE;
00424 }
