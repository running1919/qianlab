00049 /**
00050  * Do copy for 8-byte aligned data
00051  */
00052 static void AlignedCopy(uint8_t * *src,
00053                         uint8_t * *dst,
00054                         uint32_t len)
00055 {
00056     uint32_t *src32, *dst32;
00057     src32 = (uint32_t*)*src;
00058     dst32 = (uint32_t*)*dst;
00059     for (;len > 0; len -= 8) {
00060         *dst32 ++ = *src32 ++;
00061         *dst32 ++ = *src32 ++;
00062     }
00063     *src = (uint8_t*)src32;
00064     *dst = (uint8_t*)dst32;
00065 }

00070 static void UnalignedCopy(uint8_t * *src,

00072                           uint32_t  len)
00073 {
00074     for (;len > 0; len --) {

00097 {
00098     uint8_t *source;
00099     uint8_t *dest;
00100     copyFunction  *pCpy;
00101 
00102     // Check that the media is ready
00103     if (media->state != MED_STATE_READY) {
00104 
00105         TRACE_INFO("Media busy\n\r");
00106         return MED_STATUS_BUSY;
00107     }
00108 
00109     // Check that the data to read is not too big
00110     if ((length + address) > media->size) {
00111 
00112         TRACE_WARNING("RamDisk_Read: Data too big: %u, 0x%08X\n\r",
00113                       length, address);
00114         return MED_STATUS_ERROR;
00115     }
00116 
00117     // Enter Busy state
00118     media->state = MED_STATE_BUSY;
00119 
00120     // Source & Destination
00121     source = (uint8_t *)(media->blockSize
00122                                     * (address + media->baseAddress));
00123     dest   = (uint8_t *)data;

00130         pCpy(&source, &dest, media->blockSize);
