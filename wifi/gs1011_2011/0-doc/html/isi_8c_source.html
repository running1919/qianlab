00102 void ISI_CodecPathFull(void)
00103 {
00104     /* The codec path is enabled and the next frame is captured.
00105     Both codec and preview datapaths are working simultaneously */
00106     ISI->ISI_CR |= ISI_CR_ISI_CDC;
00107     /* A new request cannot be taken into account while CDC_PND bit is active in the ISI_STATUS register.*/
00108     while( (ISI->ISI_SR & ISI_SR_CDC_PND) != ISI_SR_CDC_PND);
00109     /* Both preview and codec DMA channels are operating simultaneously */
00110     ISI->ISI_CFG1 |= ISI_CFG1_FULL;

00152 /**
00153  * \brief Reset ISI
00154  */
00155 void ISI_Reset(void)
00156 {
00157     volatile uint32_t timeout = 0;
00158     if (!(ISI->ISI_SR & ISI_SR_ENABLE))
00159         return;
00160     /* Resets the image sensor interface. Finish capturing the current frame and then shut down the module.*/
00161     ISI->ISI_CR = ISI_CR_ISI_SRST | ISI_CR_ISI_DIS;
00162     /* wait Software reset has completed successfully. */
00163     while( !(ISI->ISI_SR & ISI_SR_SRST) && (timeout < 0x500000) ){
00164         timeout++;
00165     }
00166     if( timeout == 0x500000 ) {
00167         TRACE_ERROR("ISI-Reset timeout\n\r");
00168     }
00169 }
