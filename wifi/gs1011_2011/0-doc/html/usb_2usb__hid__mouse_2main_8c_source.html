00240 /**
00241  * Monitor buttons of joystick status.
00242  * \param pBtnStatus Pointer to button status bitmap.
00243  * \param pDx        Pointer to fill x value.
00244  * \param pDy        Pointer to fill y value.
00245  */
00246 static uint8_t _ButtonsMonitor(uint8_t *pBtnStatus,
00247                                int8_t *pDx,
00248                                int8_t *pDy)
00249 {
00250     uint8_t isChanged = 0;
00251 
00252   #ifdef NO_PUSHBUTTON
00253     /* - Movement W S A D */
00254     if (DBGU_IsRxReady())
00255     {
00256         uint8_t key = DBGU_GetChar();
00257         *pDx = 0; *pDy = 0;
00258         switch (key)
00259         {
00260             case 'w': case 'W':
00261                 *pDy = -SPEED_Y; isChanged = 1;
00262                 break;
00263             case 's': case 'S':
00264                 *pDy = +SPEED_Y; isChanged = 1;
00265                 break;
00266             case 'a': case 'A':
00267                 *pDx = -SPEED_X; isChanged = 1;
00268                 break;
00269             case 'd': case 'D':
00270                 *pDx = +SPEED_X; isChanged = 1;
00271                 break;
00272             default:
00273                 break;
00274         }
00275     }
00276   #else
00277     /* - Movment buttons, Joystick or Push buttons */
00278     /* Left */
00279     if (PIO_Get(&pinsJoystick[JOYSTICK_LEFT]) == 0) {
00280   
00281         *pDx = -SPEED_X;
00282         isChanged = 1;
00283     }
00284     /* Right */
00285     else if (PIO_Get(&pinsJoystick[JOYSTICK_RIGHT]) == 0) {
00286   
00287         *pDx = SPEED_X;
00288         isChanged = 1;
00289     }
00290     else {
00291         *pDx = 0;
00292     }
00293   #endif
00294 
00295     return isChanged;
00296 }
