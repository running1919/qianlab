00030 /** \addtogroup spi_at45_module SPI AT45 driver
00031  * \ingroup lib_spiflash
00032  * The Dataflash driver is based on top of the corresponding Spi driver.
00033  * A Dataflash structure instance has to be initialized using the DF_Init
00034  * function. Then basic dataflash operations can be launched using macros such
00035  * as DF_continuous_read. These macros invoke the DF_Command() function which
00036  * invokes the DPI low driver using the SPI_SendCommand() function.
00037  * Beware to compute the dataflash internal address, the dataflash sector
00038  * description must be known (DataflashDesc). Dataflash can be automatically
00039  * detected using the DF_Scan() function.
00040  *
00041  * \section Usage
00042  * <ul>
00043  * <li> Initializes an AT45 instance and configures SPI chip select pin
00044  *    using AT45_Configure(). </li>
00045  * <li> Detect DF and returns DF description corresponding to the device
00046  *    connected using AT45_FindDevice().This function shall be called by
00047  *    the application before AT45_SendCommand.</li>
00048  * <li>Sends a command to the DF through the SPI using AT45_SendCommand().
00049  *    The command is identified by its command code and the number of
00050  *    bytes to transfer.</li>
00051  *    <li>Example code for sending command to write a page to DF. </li>
00052  * \code
00053  *    // Issue a page write through buffer 1 command
00054  *     error = AT45_SendCommand(pAt45, AT45_PAGE_WRITE_BUF1, 4,
00055  *             pBuffer, size, address, 0, 0);
00056  * \endcode
00057  *    <li>Example code for sending command to read a page from DF.
00058  *       If data needs to be received, then a data buffer must be
00059  *       provided.</li>
00060  * \code
00061  *    // Issue a continuous read array command
00062  *    error = AT45_SendCommand(pAt45, AT45_CONTINUOUS_READ_LEG, 8,
00063  *            pBuffer, size, address, 0, 0);
00064  * \endcode
00065  *    <li> This function does not block; its optional callback will
00066  *       be invoked when the transfer completes.</li>
00067  * <li> Check the AT45 driver is ready or not by polling AT45_IsBusy().
00068  * </ul>
00069  * Related files :\n
00070  * \ref spi_at45.c\n
00071  * \ref spi_at45.h.\n
00072 */

00263 /**
00264  * \brief returns the At45Desc structure corresponding to the device connected.
00265  * It automatically initializes pAt45->pDesc field structure.
00266  *
00267  * \note This function shall be called by the application before AT45_SendCommand.
00268  *
00269  * \param pAt45  Pointer to the At45 instance to initialize.
00270  * \param status  Device status register value.

00272  * \return 0 if successful; Otherwise, returns AT45_ERROR_LOCK if the At45
00273  * driver is in use or AT45_ERROR_SPI if there was an error with the SPI driver.

00275 extern const At45Desc * AT45_FindDevice( At45 *pAt45, uint8_t status )
00276 {
00277     uint32_t i;
00278     uint8_t id = AT45_STATUS_ID(status);
00279 
00280     /* Check if status is all one; in which case, it is assumed that no device is connected*/
00281     if ( status == 0xFF )
00282     {
00283         return 0 ;
00284     }
00285 
00286     /* Look in device array */
00287     i = 0 ;
00288     pAt45->pDesc = 0 ;
00289     while ( (i < NUMDATAFLASH) && !(pAt45->pDesc) )
00290     {
00291         if ( at45Devices[i].id == id )
00292         {
00293             pAt45->pDesc = &(at45Devices[i]) ;
00294         }
00295         i++ ;
00296     }
00297 
00298     configuredBinaryPage = AT45_STATUS_BINARY(status);
00299 
00300     return pAt45->pDesc ;
