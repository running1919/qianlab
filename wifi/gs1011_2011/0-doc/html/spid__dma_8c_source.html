00301 /**
00302  * \brief Starts a SPI master transfer. This is a non blocking function. It will
00303  *  return as soon as the transfer is started.
00304  *
00305  * \param pSpid  Pointer to a Spid instance.
00306  * \param pCommand Pointer to the SPI command to execute.
00307  * \returns 0 if the transfer has been started successfully; otherwise returns
00308  * SPID_ERROR_LOCK is the driver is in use, or SPID_ERROR if the command is not
00309  * valid.
00310  */
00311 uint32_t SPID_SendCommand( Spid *pSpid, SpidCmd *pCommand)
00312 {
00313     Spi *pSpiHw = pSpid->pSpiHw;
00314          
00315     /* Try to get the dataflash semaphore */
00316      if (pSpid->semaphore == 0) {
00317     
00318          return SPID_ERROR_LOCK;
00319     }
00320      pSpid->semaphore--;
00321 
00322     /* Enable the SPI Peripheral */
00323     PMC_EnablePeripheral (pSpid->spiId );
00324     
00325     /* SPI chip select */
00326     SPI_ChipSelect (pSpiHw, 1 << pCommand->spiCs);
00327 
00328 #if !defined(USE_SPI_DMA)
00329     /* Initialize the callback */
00330     pSpid->pCurrentCommand = pCommand;
00331     /* Enables the SPI to transfer and receive data. */
00332     SPI_Enable (pSpiHw );
00333     {
00334     uint32_t i;
00335     /* Transfer command */
00336     for (i = 0; i < pCommand->cmdSize; i ++)
00337     {
00338         SPI_Write(pSpiHw, pCommand->spiCs, pCommand->pCmd[i]);
00339         SPI_Read(pSpiHw);
00340     }
00341     /* Transfer data */
00342     for (i = 0; i < pCommand->dataSize; i ++)
00343     {
00344         SPI_Write(pSpiHw, pCommand->spiCs, pCommand->pData[i]);
00345         pCommand->pData[i] = SPI_Read(pSpiHw);
00346     }
00347     SPI_ReleaseCS(pSpiHw);
00348     
00349     /* Unlock */
00350     pSpid->semaphore ++;
00351     
00352     /* Callback */
00353     if (pCommand->callback)
00354     {
00355         pCommand->callback(0, pCommand->pArgument);
00356     }
00357     }
00358 #else
00359     /* Initialize DMA controller using channel 0 for RX, 1 for TX. */
00360     if (_spid_configureDmaChannels(pSpid) )
00361         return SPID_ERROR_LOCK;
00362     if (_spid_configureLinkList(pSpiHw, pSpid->pDmad, pCommand))
00363         return SPID_ERROR_LOCK;
00364 
00365     // Initialize the callback
00366     pSpid->pCurrentCommand = pCommand;
00367     /* Enables the SPI to transfer and receive data. */
00368     SPI_Enable (pSpiHw );
00369 
00370     /* Start DMA 0(RX) && 1(TX) */
00371     if (DMAD_StartTransfer( pSpid->pDmad, spiDmaRxChannel )) 
00372         return SPID_ERROR_LOCK;
00373     if (DMAD_StartTransfer( pSpid->pDmad, spiDmaTxChannel )) 
00374         return SPID_ERROR_LOCK;
00375 #endif
00376     return 0;    
00377 }
