00043 /**
00044  * \brief Get the power of input, given a certain result, i.e. input^(power) = result.
00045  * \param result  a certain output we want to calculate.
00046  * \param input  the input of the power.
00047  * \return the value of "power" if succesfully find the power.
00048 */
00049 
00050 #if defined(OP_BOOTSTRAP_on)
00051 uint32_t CALPOW(uint32_t result, uint32_t input)
00052 {
00053     uint32_t i=0;
00054     while(i<32)
00055     {
00056         if(result == (input << i))
00057             return i;
00058         i++;
00059     }
00060     return 0;
00061 }
00062 #endif

00066  * \param result  a certain output we want to calculate.

00068  * \return the value of interger part of the result/input.

00070

00073 {

00075     uint32_t tmpInput=0;

00081         if(tmpInput == result)
00082             return i;
00083         else if (tmpInput > result)
00084         return (i-1);

00175 /**
00176  * \brief Translates address/size access of a NandFlashModel to block, page and
00177  * offset values. The values are stored in the provided variables if their
00178  * pointer is not 0.
00179  * \param model  Pointer to a NandFlashModel instance.
00180  * \param address  Access address.
00181  * \param size  Access size in bytes.
00182  * \param block  Stores the first accessed block number.
00183  * \param page  Stores the first accessed page number inside the first block.
00184  * \param offset  Stores the byte offset inside the first accessed page.
00185  * \return 0 if the access is correct; otherwise returns
00186  * NandCommon_ERROR_OUTOFBOUNDS.
00187 */
00188 uint8_t NandFlashModel_TranslateAccess(
00189     const struct NandFlashModel *model,
00190     uint32_t address,
00191     uint32_t size,
00192     uint16_t *block,
00193     uint16_t *page,
00194     uint16_t *offset)
00195 {
00196      /* Check that access is not too big */
00197     #if !defined(OP_BOOTSTRAP_on)
00198     if ((address + size) > NandFlashModel_GetDeviceSizeInBytes(model)) {
00199 
00200         TRACE_DEBUG("NandFlashModel_TranslateAccess: out-of-bounds access.\n\r");
00201         return NandCommon_ERROR_OUTOFBOUNDS;
00202     }
00203     #endif
00204 
00205     /* Get Nand info */
00206     uint32_t blockSize = NandFlashModel_GetBlockSizeInBytes(model);
00207     uint32_t pageSize = NandFlashModel_GetPageDataSize(model);
00208 
00209     /* Translate address */
00210     #if !defined(OP_BOOTSTRAP_on)
00211     uint16_t tmpBlock = address / blockSize;
00212     address -= tmpBlock * blockSize;
00213     uint16_t tmpPage = address / pageSize;
00214     address -= tmpPage * pageSize;
00215     uint16_t tmpOffset = address;
00216     #else
00217     uint16_t tmpBlock = CALINT(address, blockSize);
00218     address -= tmpBlock * blockSize;
00219     uint16_t tmpPage = CALINT(address, pageSize);
00220     address -= tmpPage * pageSize;
00221     uint16_t tmpOffset= address;
00222     #endif
00223 
00224     // Save results
00225     if (block) {
00226 
00227         *block = tmpBlock;
00228     }
00229     if (page) {
00230 
00231         *page = tmpPage;
00232     }
00233     if (offset) {
00234 
00235         *offset = tmpOffset;
00236     }
00237 
00238     return 0;
00239 }
