00069 /**
00070  * \brief Compute the CRC
00071  */
00072 static uint16_t _GetCRC(uint8_t bByte, uint16_t wCrc)
00073 {
00074     int32_t cnt;
00075     uint8_t  newBit;
00076     for (cnt = 7; cnt >= 0; cnt --)
00077     {
00078         newBit = ((wCrc >> 15) & 0x1) ^ ((bByte >> cnt) & 0x1);
00079         wCrc <<= 1;
00080         if (newBit) wCrc ^= (0x1021);
00081     }
00082     return wCrc;
00083     
00084 }
