00031 /** \cond can
00032  *  \page can  CAN Example
00033  *
00034  *  \section Purpose
00035  *
00036  *  The CAN example demostrates how to use CAN peripheral.
00037  *
00038  *  \section Requirements
00039  *
00040  *  This package can be used with AT91SAM9xx5 evaluation kits with AT91SAM9X25 
00041  *  or AT91SAM9X35.
00042  *
00043  *  \subsection hw_setup Hardware Connection Setup
00044  *
00045  *  - JP11 must be openned to enable CAN0 transceiver.
00046  *  - USART0 must be used as debug port since CAN0 pins are conflicted with DBGU.
00047  *  - Connect CON2 and CON3 so CAN0 and CAN1 are connected.
00048  *
00049  *  \section Description
00050  *  
00051  *  The application is aimed to demonstrate how to use CAN. The example tests
00052  *  following CAN operations:<br>
00053  *  <ul>
00054  *    <li> Simple CAN test. CAN1 Mailbox 5 sends data to CAN0 Mailbox 1.
00055  *    </li>
00056  *    <li> Messages to 1 Mailbox test. CAN1 Mailbox 5 and 6 send data,
00057  *         (6 goes first), CAN0 Mailbox 2 receives them, but last one
00058  *         is discarded.
00059  *    </li>
00060  *    <li> Messages to 1 Mailbox test. CAN1 Mailbox 6 and 5 sends data in
00061  *         sequence, with ID 0x40 and 0x41 that both can accepted by CAN0
00062  *         Mailbox 3. The last data overwrites prievious one.
00063  *    </li>
00064  *    <li> Remote data request test. CAN1 Mailbox 5 sends remote request to
00065  *         CAN0 Mailbox 4 to get response data.
00066  *    </li>
00067  *  </ul><br>
00068  *
00069  *  \section Usage
00070  *
00071  *  -# Build the program and download it inside the evaluation board. Please
00072  *     refer to the
00073  *     <a href="http://www.atmel.com/dyn/resources/prod_documents/6421B.pdf">
00074  *     SAM-BA User Guide</a>, the
00075  *     <a href="http://www.atmel.com/dyn/resources/prod_documents/doc6310.pdf">
00076  *     GNU-Based Software Development</a>
00077  *     application note or to the
00078  *     <a href="ftp://ftp.iar.se/WWWfiles/arm/Guides/EWARM_UserGuide.ENU.pdf">
00079  *     IAR EWARM User Guide</a>,
00080  *     depending on your chosen solution.
00081  *  -# On the computer, open and configure a terminal application
00082  *     (e.g. HyperTerminal on Microsoft Windows) with these settings:
00083  *    - 115200 bauds
00084  *    - 8 bits of data
00085  *    - No parity
00086  *    - 1 stop bit
00087  *    - No flow control
00088  *  -# Start the application.
00089  *  -# In the terminal window, the
00090  *     following text should appear (values depend on the board and chip used):
00091  *     \code
00092  *      -- CAN Example xxx --
00093  *      -- AT91xxxxxx-xx
00094  *      -- Compiled: xxx xx xxxx xx:xx:xx --
00095  *     \endcode
00096  *  -# Then the tests results appear on the terminal.
00097  *
00098  *  \section References
00099  *  - can/main.c
00100  *  - pio.h
00101  *  - pio_it.h
00102  */

00258 /**
00259  * Configure for RX/Producer tests.
00260  * MB1 for simple receiving test (ID 0x30).
00261  * MB2 for rx discarding test (ID 0x31).
00262  * MB3 for rx overwriting test (ID 0x32).
00263  * MB4 for message pull test (ID 0x33).
00264  * \param nbCan CAN number.
00265  */
00266 static void _CanRxInitialize(uint8_t nbCan)
00267 {
00268     sCand *pCand = &cand[nbCan];
00269     sCandMbCfg candCfg;
00270 
00271     /* MB1: simple RX */
00272     /* ID: 0x31 */
00273     /* Right data: 0x00FF1122, 4 bytes */
00274     candCfg.bMsgType = CAN_MMR_MOT_MB_RX >> CAN_MMR_MOT_Pos;
00275     candCfg.bTxPriority = 0;
00276     candCfg.dwMsgMask = CAN_MID_MIDvA_Msk;
00277     rxSimple[nbCan].bMailbox = 1;
00278     rxSimple[nbCan].dwMsgID = CAN_MID_MIDvA(0x31);
00279     CAND_ConfigureTransfer(pCand, &candCfg, &rxSimple[nbCan]);
00280 
00281     /* MB2: RX 2 messages, the last one discarded */
00282     /* ID: 0x32 */
00283     /* Right data: 0x01FF1111, 4 bytes */
00284     rxDiscard[nbCan].bMailbox = 2;
00285     rxDiscard[nbCan].dwMsgID = CAN_MID_MIDvA(0x32);
00286     CAND_ConfigureTransfer(pCand, &candCfg, &rxDiscard[nbCan]);
00287 
00288     /* MB3: RX 2 messages, the last one overwrite first */
00289     /* ID: 0x40 or 0x41 */
00290     /* Right data: 0x02FF2222 0x02FF2345, 8 bytes */
00291     candCfg.bMsgType = CAN_MMR_MOT_MB_RX_OVERWRITE >> CAN_MMR_MOT_Pos;
00292     candCfg.dwMsgMask = CAN_MID_MIDvA_Msk & ~(CAN_MID_MIDvA(1));
00293     rxOverwr[nbCan].bMailbox = 3;
00294     rxOverwr[nbCan].dwMsgID = CAN_MID_MIDvA(0x40);
00295     CAND_ConfigureTransfer(pCand, &candCfg, &rxOverwr[nbCan]);
00296 
00297     /* MB4: waiting Remote Frame (for pulling mode). */
00298     /* ID: 0x34 */
00299     /* Send data: 0x03EF, 2 bytes */
00300     candCfg.bMsgType = CAN_MMR_MOT_MB_PRODUCER >> CAN_MMR_MOT_Pos;
00301     candCfg.dwMsgMask = CAN_MID_MIDvA_Msk;
00302     rxTx[nbCan].bMailbox = 4;
00303     rxTx[nbCan].dwMsgID = CAN_MID_MIDvA(0x34);
00304     rxTx[nbCan].bMsgLen = 2;
00305     rxTx[nbCan].msgData[0] = 0x03EF;
00306     CAND_ConfigureTransfer(pCand, &candCfg, &rxTx[nbCan]);
00307 
00308     /* Start transfers */
00309     CAND_StartTransfers(pCand, CAN_SR_MB1 | CAN_SR_MB2 | CAN_SR_MB4);
00310 }
