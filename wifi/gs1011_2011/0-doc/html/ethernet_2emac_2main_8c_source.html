00030 /**
00031  *  \cond emac
00032  *  \page emac EMAC Example
00033  * 
00034  *  \section Purpose
00035  * 
00036  *  This example uses the Ethernet MAC (EMAC) and the on-board Ethernet
00037  *  transceiver available on Atmel evaluation kits. It enables the device to
00038  *  respond to a ping command sent by a host computer.
00039  * 
00040  *  \section Requirements
00041  * 
00042  *  - AT91 microcontrollers with EMAC feature.
00043  *  - On-board ethernet interface. 
00044  * 
00045  *  \section Description
00046  * 
00047  *  Upon startup, the program will configure the EMAC with a default IP and
00048  *  MAC address and then ask the transceiver to auto-negotiate the best mode
00049  *  of operation. Once this is done, it will start to monitor incoming packets
00050  *  and processing them whenever appropriate.
00051  *
00052  *  The basic will only answer to two kinds of packets:
00053  * 
00054  *  - It will reply to ARP requests with its MAC address,
00055  *  - and to ICMP ECHO request so the device can be PING'ed.
00056  * 
00057  *  You may use 'ping' command to check if the board responds correctly to
00058  *  ping requests.
00059  * 
00060  *  \section Usage
00061  * 
00062  *  -# Build the program and download it inside the evaluation board. Please
00063  *     refer to the
00064  *     <a href="http://www.atmel.com/dyn/resources/prod_documents/6421B.pdf">
00065  *     SAM-BA User Guide</a>, the
00066  *     <a href="http://www.atmel.com/dyn/resources/prod_documents/doc6310.pdf">
00067  *     GNU-Based Software Development</a>
00068  *     application note or to the
00069  *     <a href="ftp://ftp.iar.se/WWWfiles/arm/Guides/EWARM_UserGuide.ENU.pdf">
00070  *     IAR EWARM User Guide</a>,
00071  *     depending on your chosen solution.
00072  *  -# On the computer, open and configure a terminal application
00073  *     (e.g. HyperTerminal on Microsoft Windows) with these settings:
00074  *    - 115200 bauds
00075  *    - 8 bits of data
00076  *    - No parity
00077  *    - 1 stop bit
00078  *    - No flow control
00079  *  -# Connect an Ethernet cable between the evaluation board and the network.
00080  *      The board may be connected directly to a computer; in this case,
00081  *      make sure to use a cross/twisted wired cable such as the one provided
00082  *      with the evaluation kit.
00083  *  -# Start the application. It will display the following message on the DBGU:
00084  *     \code
00085  *      -- EMAC Example xxx --
00086  *      -- AT91xxxxxx-xx
00087  *      -- Compiled: xxx xx xxxx xx:xx:xx --
00088  *      MAC 00:45:56:78:9a:bc
00089  *      IP 10.217.3.168
00090  *     \endcode
00091  *  -# The program will then auto-negotiate the mode of operation and start
00092  *     receiving packets, displaying feedback on the DBGU. To display additional
00093  *     information, press any key in the terminal application.
00094  *  -# To test that the board responds to ICMP ECHO requests, type the following
00095  *     command line in a shell:
00096  *      \code
00097  *       ping 10.217.3.168
00098  *      \endcode
00099  *     Response to 'ping' cmd will appear in the shell.
00100  *  -# Press any key in the hyperterminal, a statistic of the EMAC communication will be displayed.
00101  * 
00102  *  \note
00103  *  Make sure the IP adress of the device(EK board) and the computer are in the same network.
00104  *
00105  *  \section References
00106  *  - emac/main.c
00107  *  - macb.h
00108  *  - emacd.h
00109  *  - emac.h
00110  */

00279  * Process the received ARP packet
00280  * Just change address and send it back
00281  * \param pData The data to process
00282  * \param size The data size
00283  */
00284 static void arp_process_packet(uint8_t* pData, uint32_t size)
00285 {
00286     uint32_t i;
00287     uint8_t emac_rc = EMACD_OK;
00288 
00289     PEthHeader   pEth = (PEthHeader)pData;
00290     PArpHeader   pArp = (PArpHeader)(pData + 14 + EMAC_RCV_OFFSET);
00291 
00292     if (SWAP16(pArp->ar_op) == ARP_REQUEST)
00293     {
00294         /* ARP REPLY operation */
00295         pArp->ar_op =  SWAP16(ARP_REPLY);
00296 
00297         /* Fill the dest address and src address */
00298         for (i = 0; i <6; i++)
00299         {
00300             /* swap ethernet dest address and ethernet src address */
00301             pEth->et_dest[i] = pEth->et_src[i];
00302             pEth->et_src[i]  = MacAddress[i];
00303             pArp->ar_tha[i]  = pArp->ar_sha[i];
00304             pArp->ar_sha[i]  = MacAddress[i];
00305         }
00306         /* swap sender IP address and target IP address */
00307         for (i = 0; i<4; i++)
00308         {
00309             pArp->ar_tpa[i] = pArp->ar_spa[i];
00310             pArp->ar_spa[i] = IpAddress[i];
00311         }
00312         emac_rc = EMACD_Send(&gEmacd,
00313                              (pData + EMAC_RCV_OFFSET),
00314                              size,
00315                              NULL);
00316         if (emac_rc != EMACD_OK)
00317         {
00318             printf("E: ARP Send - 0x%x\n\r", emac_rc);
00319         }
00320     }
