00188 /**
00189  * Interpolates the provided raw measurements using the previously calculated
00190  * slope. The resulting x and y coordinates are stored in an array.
00191  * \param pData  Raw measurement data, as returned by TSD_GetRawMeasurement().
00192  * \param pPoint  Array in which x and y will be stored.
00193  */
00194 void TSDCom_InterpolateMeasurement(const uint32_t *pData, uint32_t *pPoint)
00195 {
00196     pPoint[0] = calibrationPoints[0].x
00197                 - (((int32_t) calibrationPoints[0].data[0] - (int32_t) pData[0]) * 1024)
00198                 / xSlope;
00199 
00200     pPoint[1] = calibrationPoints[0].y
00201                 - (((int32_t) calibrationPoints[0].data[1] - (int32_t) pData[1]) * 1024)
00202                 / ySlope;
00203     /* Is pPoint[0] negative ? */
00204     if(pPoint[0] & 0x80000000)        pPoint[0] = 0;
00205     /* Is pPoint[0] bigger than the LCD width ? */
00206     if(pPoint[0] > BOARD_LCD_WIDTH)   pPoint[0] = BOARD_LCD_WIDTH;
00207     /* Is pPoint[1] negative ? */
00208     if(pPoint[1] & 0x80000000)        pPoint[1] = 0;
00209     /* Is pPoint[1] bigger than the LCD width ? */
00210     if(pPoint[1] > BOARD_LCD_HEIGHT)  pPoint[1] = BOARD_LCD_HEIGHT;
00211 }

00213 /**
00214  * Performs the calibration process using the provided buffer to display
00215  * information.
00216  * \param pLcdBuffer  LCD buffer to display.
00217  * \return True if calibration was successful; otherwise false.
00218  */
00219 uint8_t TSDCom_Calibrate(void)
00220 {
00221     volatile uint32_t i; // to keep the tempo with gcc code optimisation
00222     int32_t slope1, slope2;
00223     CalibrationPoint measuredPoint;
00224     uint8_t xOk, yOk;
00225     int32_t xDiff, yDiff;
00226     uint32_t strX = BOARD_LCD_WIDTH / 2 - 75, strY = 60;
00227     uint32_t strW, strH;
00228 
00229     LCDD_GetStringSize("P", &strW, &strH);
00230     /* Calibration setup */
00231     LCDD_Fill(COLOR_WHITE);
00232     LCDD_DrawString(strX, strY, strTitle, COLOR_BLACK);
00233     LCDD_DrawString(strX - 2*strW, strY + 3*strH,
00234         " Touch the dots to\ncalibrate the screen", COLOR_DARKBLUE);
00235 
00236     /* Calibration points */
00237     for (i=0; i < 4; i++) {
00238 
00239         DrawCalibrationPoint(&calibrationPoints[i]);
00240 
00241         /* Wait for touch & end of conversion */
00242         TSD_WaitPenPressed();
00243         TSD_GetRawMeasurement(calibrationPoints[i].data);
00244         ClearCalibrationPoint(&calibrationPoints[i]);
00245 
00246         /* Wait for contact loss */
00247         TSD_WaitPenReleased();
00248 
00249         TRACE_INFO("P%d: (%d,%d)\n\r",
00250             i, calibrationPoints[i].data[0], calibrationPoints[i].data[1]);
00251     }
00252 
00253     /* Calculate slopes using the calibration data
00254      * Theory behind those calculations:
00255      *   - We suppose the touchscreen measurements are linear, so the following equations are true (simple
00256      *     linear regression) for any two 'a' and 'b' points of the screen:
00257      *       dx = (a.data[0] - b.data[0]) / (a.x - b.x)
00258      *       dy = (a.data[1] - b.data[1]) / (a.y - b.y)
00259      *
00260      *   - We calculate dx and dy (called xslope and yslope here) using the calibration points.
00261      *
00262      *   - We can then use dx and dy to infer the position of a point 'p' given the measurements performed
00263      *     by the touchscreen ('c' is any of the calibration points):
00264      *       dx = (p.data[0] - c.data[0]) / (p.x - c.x)
00265      *       dy = (p.data[1] - c.data[1]) / (p.y - c.y)
00266      *     Thus:
00267      *       p.x = c.x - (p.data[0] - c.data[0]) / dx
00268      *       p.y = c.y - (p.data[1] - c.data[1]) / dy
00269      *
00270      *   - Since there are four calibration points, dx and dy can be calculated twice, so we average
00271      *     the two values.
00272      */
00273     slope1 = ((int32_t) calibrationPoints[0].data[0]) - ((int32_t) calibrationPoints[1].data[0]);
00274     slope1 *= 1024;
00275     slope1 /= ((int32_t) calibrationPoints[0].x) - ((int32_t) calibrationPoints[1].x);
00276     slope2 = ((int32_t) calibrationPoints[2].data[0]) - ((int32_t) calibrationPoints[3].data[0]);
00277     slope2 *= 1024;
00278     slope2 /= ((int32_t) calibrationPoints[2].x) - ((int32_t) calibrationPoints[3].x);
00279     xSlope = (slope1 + slope2) / 2;
00280 
00281     slope1 = ((int32_t) calibrationPoints[0].data[1]) - ((int32_t) calibrationPoints[2].data[1]);
00282     slope1 *= 1024;
00283     slope1 /= ((int32_t) calibrationPoints[0].y) - ((int32_t) calibrationPoints[2].y);
00284     slope2 = ((int32_t) calibrationPoints[1].data[1]) - ((int32_t) calibrationPoints[3].data[1]);
00285     slope2 *= 1024;
00286     slope2 /= ((int32_t) calibrationPoints[1].y) - ((int32_t) calibrationPoints[3].y);
00287     ySlope = (slope1 + slope2) / 2;
00288 
00289     TRACE_INFO("Slope: %d, %d\n\r", xSlope, ySlope);
00290 
00291     /* Test point */
00292     CLEAR_STRING();
00293     LCDD_DrawString(strX, strY, strTitle, COLOR_BLACK);
00294     LCDD_DrawString(strX - 2*strW, strY + 3*strH,
00295         " Touch the point to\nvalidate calibration", COLOR_DARKBLUE);
00296     DrawCalibrationPoint(&testPoint);
00297 
00298     /* Wait for touch & end of conversion */
00299     TSD_WaitPenPressed();
00300 
00301     TSD_GetRawMeasurement(measuredPoint.data);
00302     TSDCom_InterpolateMeasurement(measuredPoint.data, (uint32_t *) &measuredPoint);
00303     DrawCalibrationPoint(&measuredPoint);
00304 
00305     /* Check resulting x and y */
00306     xDiff = (int32_t) measuredPoint.x - (int32_t) testPoint.x;
00307     yDiff = (int32_t) measuredPoint.y - (int32_t) testPoint.y;
00308     xOk = (xDiff >= -POINTS_MAX_XERROR) && (xDiff <= POINTS_MAX_XERROR);
00309     yOk = (yDiff >= -POINTS_MAX_YERROR) && (yDiff <= POINTS_MAX_YERROR);
00310 
00311     /* Wait for contact loss */
00312     TSD_WaitPenReleased();
00313 
00314     TRACE_INFO("TP: %d, %d -> %d, %d\n\r",
00315         measuredPoint.data[0], measuredPoint.data[1],
00316         measuredPoint.x, measuredPoint.y);
00317 
00318     /* Check calibration result */
00319     if (xOk && yOk) {
00320 
00321         bCalibrationOk = 1;
00322         CLEAR_STRING();
00323         LCDD_DrawString(strX, strY, strTitle, COLOR_BLACK);
00324         LCDD_DrawString(strX + 3*strW, strY + 2*strH, "Success !", COLOR_GREEN);
00325 
00326     }
00327     else {
00328 
00329         bCalibrationOk = 0;
00330         CLEAR_STRING();
00331         LCDD_DrawString(strX, strY, strTitle, COLOR_BLACK);
00332         LCDD_DrawString(strX + strW, strY + 2*strH, "Error too big", COLOR_RED);
00333 
00334         TRACE_WARNING("X %u, Y %u; Diff %d, %d\n\r",
00335             measuredPoint.x, measuredPoint.y, xDiff, yDiff);
00336     }
00337 
00338     /* Slight delay */
00339     for (i = 0; i < DELAY_RESULT_DISPLAY; i++);
00340 
00341     return (xOk && yOk);
00342 }
