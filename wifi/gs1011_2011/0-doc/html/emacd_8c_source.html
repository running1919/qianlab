00054 /** Return count in buffer */
00055 #define CIRC_CNT(head,tail,size) (((head) - (tail)) % (size))
00056 
00057 /** Return space available, 0..size-1.
00058     We always leave one free char as a completely full buffer 
00059     has head == tail, which is the same as empty */
00060 #define CIRC_SPACE(head,tail,size) CIRC_CNT((tail),((head)+1),(size))
00061 
00062 /** Return count up to the end of the buffer.  
00063    Carefully avoid accessing head and tail more than once,
00064    so they can change underneath us without returning inconsistent results */
00065 #define CIRC_CNT_TO_END(head,tail,size) \
00066    ({int end = (size) - (tail); \
00067      int n = ((head) + end) % (size); \
00068      n < end ? n : end;})
00069 
00070 /** Return space available up to the end of the buffer */
00071 #define CIRC_SPACE_TO_END(head,tail,size) \
00072    ({int end = (size) - 1 - (head); \
00073      int n = (end + (tail)) % (size); \
00074      n <= end ? n : end+1;})
00075 
00076 /** Increment head or tail */
00077 #define CIRC_INC(headortail,size) \
00078         headortail++;             \
00079         if(headortail >= size) {  \
00080             headortail = 0;       \
00081         }
00082 /** Circular buffer is empty ? */
00083 #define CIRC_EMPTY(head, tail)     (head == tail)
00084 /** Clear circular buffer */
00085 #define CIRC_CLEAR(head, tail)     (head = tail = 0)
