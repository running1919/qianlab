00246     pEmac->EMAC_NCFGR |=  (EMAC_NCFGR_CLK_Msk & ((bCLK) << EMAC_NCFGR_CLK_Pos));
00247 
00248     return 1;
00249 }

00436 /**
00437  * Execute PHY maintanance command
00438  */
00439 void EMAC_PHYMaintain(Emac      *pEmac,
00440                       uint8_t   bPhyAddr,
00441                       uint8_t   bRegAddr,
00442                       uint8_t   bRW,
00443                       uint16_t  wData)
00444 {
00445     /* Wait until bus idle */
00446     while((pEmac->EMAC_NSR & EMAC_NSR_IDLE) == 0);
00447     /* Write maintain register */
00448     pEmac->EMAC_MAN = EMAC_MAN_CODE(10) | EMAC_MAN_SOF(0x1)
00449                     | EMAC_MAN_PHYA(bPhyAddr)
00450                     | EMAC_MAN_REGA(bRegAddr)
00451                     | EMAC_MAN_RW((bRW ? 0x2 : 0x1))
00452                     | EMAC_MAN_DATA(wData)
00453                     ;
00454 }
