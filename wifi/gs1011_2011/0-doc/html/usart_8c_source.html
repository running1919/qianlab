00076 /**
00077  * \brief Configures an USART peripheral with the specified parameters.
00078  *
00079  *
00080  *  \param usart  Pointer to the USART peripheral to configure.
00081  *  \param mode  Desired value for the USART mode register (see the datasheet).
00082  *  \param baudrate  Baudrate at which the USART should operate (in Hz).
00083  *  \param masterClock  Frequency of the system master clock (in Hz).
00084  */
00085 void USART_Configure(Usart *usart,
00086                      uint32_t mode,
00087                      uint32_t baudrate,
00088                      uint32_t masterClock)
00089 {
00090     /* Reset and disable receiver & transmitter*/
00091     usart->US_CR = US_CR_RSTRX | US_CR_RSTTX
00092                    | US_CR_RXDIS | US_CR_TXDIS;
00093 
00094     /* Configure mode*/
00095     usart->US_MR = mode;
00096 
00097     /* Configure baudrate*/
00098     /* Asynchronous, no oversampling*/
00099     if (((mode & US_MR_SYNC) == 0)
00100         && ((mode & US_MR_OVER) == 0))
00101     {
00102         usart->US_BRGR = (masterClock / baudrate) / 16;
00103     }
00104     if( ((mode & US_MR_USART_MODE_SPI_MASTER) == US_MR_USART_MODE_SPI_MASTER)
00105      || ((mode & US_MR_SYNC) == US_MR_SYNC))
00106     {
00107         if( (mode & US_MR_USCLKS_Msk) == US_MR_USCLKS_MCK)
00108         {
00109             usart->US_BRGR = masterClock / baudrate;
00110         }
00111         else
00112         {
00113             if ( (mode & US_MR_USCLKS_DIV) == US_MR_USCLKS_DIV)
00114             {
00115                 usart->US_BRGR = masterClock / baudrate / 8;
00116             }
00117         }
00118     }
00119     /* TODO other modes*/
00120 }
