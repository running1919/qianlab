00313 /**
00314  * Parse descriptors: Interface, Interrupt IN/OUT.
00315  * \param desc Pointer to descriptor list.
00316  * \param arg  Argument, pointer to HIDDParseData instance.
00317  */
00318 static uint32_t HIDDFunction_Parse(USBGenericDescriptor * pDesc,
00319                                    HIDDParseData * pArg)
00320 {
00321     /* Find HID Interface */
00322     if (pArg->pIfDesc == 0) {
00323         if (pDesc->bDescriptorType == USBGenericDescriptor_INTERFACE) {
00324             USBInterfaceDescriptor *pIf = (USBInterfaceDescriptor*)pDesc;
00325             /* Right interface for HID:
00326                    HID Class + at least 1 endpoint */
00327             if (pIf->bInterfaceClass == HIDInterfaceDescriptor_CLASS
00328                 && pIf->bNumEndpoints >= 1) {
00329                 /* Obtain new interface setting */
00330                 if (pArg->pHidd->bInterface == 0xFF) {
00331                     pArg->pHidd->bInterface = pIf->bInterfaceNumber;
00332                     pArg->pIfDesc = pIf;
00333                 }
00334                 /* Find specific interface setting */
00335                 else if (pArg->pHidd->bInterface == pIf->bInterfaceNumber) {
00336                     pArg->pIfDesc = pIf;
00337                 }
00338             }
00339         }
00340     }
00341     /* Interface end */
00342     else {
00343         /* Start another interface ? */
00344         if (pDesc->bDescriptorType == USBGenericDescriptor_INTERFACE) {
00345             /* Terminate the parse */
00346             return USBRC_PARTIAL_DONE;
00347         }
00348         /* Parse HID descriptor */
00349         else if (pDesc->bDescriptorType == HIDGenericDescriptor_HID) {
00350             pArg->pHidd->pHidDescriptor = (HIDDescriptor*)pDesc;
00351         }
00352         /* Parse endpoints */
00353         else if (pDesc->bDescriptorType == USBGenericDescriptor_ENDPOINT) {
00354             USBEndpointDescriptor *pEp = (USBEndpointDescriptor*)pDesc;
00355             if (pEp->bEndpointAddress & 0x80)
00356                 pArg->pHidd->bPipeIN = pEp->bEndpointAddress & 0x7F;
00357             else
00358                 pArg->pHidd->bPipeOUT = pEp->bEndpointAddress;
00359         }
00360 
00361         /* Check if all data is OK */
00362         if (pArg->pHidd->bInterface  != 0xFF
00363             && pArg->pHidd->bPipeIN  != 0xFF
00364             && pArg->pHidd->bPipeOUT != 0xFF)
00365             return USBRC_FINISHED;
00366     }
00367     return 0;
00368 }

00555 /**
00556  * Handles HID-specific SETUP request sent by the host.
00557  * \param pHidd Pointer to HIDDFunction instance.
00558  * \param request Pointer to a USBGenericRequest instance
00559  */
00560 uint32_t HIDDFunction_RequestHandler(HIDDFunction *pHidd,
00561                                      const USBGenericRequest *request)
00562 {
00563     uint32_t reqCode = (request->bmRequestType << 8)
00564                      | (request->bRequest);
00565 
00566     switch (reqCode) {
00567     /* Get_Descriptor */
00568     case USBGenericRequest_GETDESCRIPTOR|(0x81<<8):
00569         return HIDDFunction_GetDescriptor(
00570                     pHidd,
00571                     USBGetDescriptorRequest_GetDescriptorType(request),
00572                     USBGenericRequest_GetLength(request));
00573     /* Clear_Feature (EP) */
00574     case USBGenericRequest_CLEARFEATURE|(0x02<<8):
00575         if (USBFeatureRequest_GetFeatureSelector(request)
00576             == USBFeatureRequest_ENDPOINTHALT) {
00577             uint8_t ep = USBGenericRequest_GetEndpointNumber(request);
00578             if (USBD_IsHalted(ep)) {
00579                 /* Unhalt EP */
00580                 USBD_Unhalt(ep);
00581                 /* Restart Polling OUT */
00582                 if (ep == pHidd->bPipeOUT) {
00583                     HIDDFunction_StartPollingOutputs(pHidd);
00584                 }
00585                 /* and send a zero-length packet */
00586                 USBD_Write(0, 0, 0, 0, 0);
00587             }
00588             break; /* Handled success */
00589         }
00590         return USBRC_PARAM_ERR;
00591     /* Set_Descriptor */
00592     case USBGenericRequest_SETDESCRIPTOR|(0x01<<8):
00593         /* Optional, not implemented */
00594         USBD_Stall(0);
00595         break;
00596     /* Get_Idle */
00597     case (0xa1<<8)|HIDGenericRequest_GETIDLE:
00598         HIDDFunction_GetIdle(pHidd,
00599                              HIDReportRequest_GetReportId(request));
00600         break;
00601     /* Set_Idle */
00602     case (0x21<<8)|HIDGenericRequest_SETIDLE:
00603         HIDDFunction_SetIdle(pHidd,
00604                              HIDReportRequest_GetReportId(request),
00605                              HIDIdleRequest_GetIdleRate(request));
00606         break;
00607     /* Get_Report */
00608     case (0xa1<<8)|HIDGenericRequest_GETREPORT:
00609         HIDDFunction_GetReport(pHidd,
00610                                HIDReportRequest_GetReportType(request),
00611                                HIDReportRequest_GetReportId(request),
00612                                USBGenericRequest_GetLength(request));
00613         break;
00614     /* Set_Report */
00615     case (0x21<<8)|HIDGenericRequest_SETREPORT:
00616         HIDDFunction_SetReport(pHidd,
00617                                HIDReportRequest_GetReportType(request),
00618                                HIDReportRequest_GetReportId(request),
00619                                USBGenericRequest_GetLength(request));
00620         break;
00621     /* Get_Protocol */
00622     case (0xa1<<8)|HIDGenericRequest_SETPROTOCOL:
00623         pHidd->bProtocol = request->wValue;
00624         USBD_Write(0, 0, 0, 0, 0);
00625         break;
00626     /* Set_Protocol */
00627     case (0x21<<8)|HIDGenericRequest_GETPROTOCOL:
00628         USBD_Write(0, &pHidd->bProtocol, 1, 0, 0);
