00103 /**
00104  * \brief Get peripheral identifier coded for hardware handshaking interface
00105  *
00106  * \param bDmac      DMA Controller number.
00107  * \param bPeriphID  Peripheral ID.
00108  * \param bTransfer  Transfer type 0: Tx, 1 :Rx.
00109  * \return 0-15 peripheral identifier coded.
00110  *         0xff : no associated peripheral identifier coded.
00111  */
00112 uint8_t DMAIF_Get_ChannelNumber (uint8_t bDmac,
00113                                  uint8_t bPeriphID, 
00114                                  uint8_t bTransfer)
00115 {
00116     uint8_t i;
00117     for (i = 0; i < NUMPERIPHERAL; i++)
00118     {
00119         if ((dmaHwIf[i].bDmac == bDmac) && (dmaHwIf[i].bPeriphID == bPeriphID) && (dmaHwIf[i].bTransfer == bTransfer))
00120         {
00121             return dmaHwIf[i].bIfID;
00122         }
00123     }
00124     return 0xff;
00125 }

00127 /**
00128  * \brief Check if the given DMAC has associated peripheral identifier coded by 
00129  * the given  peripheral.
00130  *
00131  * \param bDmac      DMA Controller number.
00132  * \param bPeriphID  Peripheral ID (0xff : memory only).
00133  * \return 1:  Is a validated peripher. 0: no associated peripheral identifier coded.
00134  */
00135 uint8_t DMAIF_IsValidatedPeripherOnDma( uint8_t bDmac, uint8_t bPeriphID)
00136 {
00137     uint8_t i;
00138     /* It is always validated when transfer to memory */
00139     if (bPeriphID == 0xFF) {
00140         return 1;
00141     }
00142     for (i = 0; i < NUMPERIPHERAL; i++)
00143     {
00144         if ((dmaHwIf[i].bDmac == bDmac) && (dmaHwIf[i].bPeriphID == bPeriphID))
00145         {
00146             return 1;
00147         }
00148     }
00149     return 0;
00150 }
