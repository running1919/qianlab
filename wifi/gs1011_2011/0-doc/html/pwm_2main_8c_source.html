00114 /**
00115  * \brief Application entry point for PWM example.
00116  *
00117  * Outputs a PWM on LED1 & LED2. 
00118  * \return Unused (ANSI-C compatibility).
00119  */
00120 int main(void)
00121 {
00122     uint32_t mode;
00123 
00124     /* Disable watchdog */
00125     WDT_Disable( WDT ) ;
00126 
00127     /* Output example information */
00128     printf("-- PWM Example %s --\n\r", SOFTPACK_VERSION);
00129     printf("-- %s\n\r", BOARD_NAME);
00130     printf("-- Compiled: %s %s --\n\r", __DATE__, __TIME__);
00131 
00132     /* PIO configuration */
00133     PIO_Configure(pins, PIO_LISTSIZE(pins));
00134 
00135     /* Enable PWM peripheral clock */
00136     PMC_EnablePeripheral(ID_PWM);
00137 
00138     /* Set clock A and clock B */
00139     mode = PWM_MR_PREB_MCKDIV1024 | (128<<PWM_MR_DIVB_Pos) | 
00140            PWM_MR_PREA_MCKDIV1024 | (128<<PWM_MR_DIVA_Pos);
00141     PWMC_ConfigureClocks(PWM, mode);
00142 
00143     /* Configure PWM channel0 for LED0  */
00144     PWMC_DisableChannel(PWM, 0);
00145     mode=PWM_CMR_CPRE_CLKA;
00146     PWMC_ConfigureChannel(PWM, 0, mode);
00147     PWMC_SetPeriod(PWM, 0, 0x0400);
00148     PWMC_SetDutyCycle(PWM, 0, 0x0200);
00149     PWMC_EnableChannel(PWM, 0);
00150 
00151     /* Configure PWM channel1 for LED1 */
00152     PWMC_DisableChannel(PWM, 1);
00153     mode = PWM_CMR_CPD | PWM_CMR_CPOL | PWM_CMR_CALG | PWM_CMR_CPRE_CLKB;
00154     PWMC_ConfigureChannel(PWM, 1, mode);
00155     PWMC_SetPeriod(PWM, 1, 0x0800);
00156     PWMC_SetDutyCycle(PWM, 1, 0x0200);
00157     PWMC_EnableChannel(PWM, 1);    
00158 
00159     while (1) 
00160     {        
00161     }
00162 }
