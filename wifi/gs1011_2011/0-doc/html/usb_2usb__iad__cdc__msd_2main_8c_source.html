00459 /**
00460  * Handles interrupts coming from Timer #0.
00461  */
00462 static void TC0_IrqHandler(void)
00463 {
00464     Tc *pTc0 = TC0;
00465     sDmad *pDmad = &dmad;
00466     uint8_t size;
00467     uint32_t nextBuffer = 1 - usartCurrentBuffer;
00468     uint32_t dmaChannel = usartDmaRxChannel;
00469     uint8_t  iController = (dmaChannel >> 8);
00470     uint8_t  iChannel    = (dmaChannel) & 0xFF;
00471     volatile uint32_t status = pTc0->TC_CHANNEL[0].TC_SR;
00472 
00473     if ((status & TC_SR_CPCS) != 0) {
00474 
00475         /* MSD Timer */
00476         if (-- msdDelay == 0)
00477         {
00478             msdRefresh = 1;
00479             msdDelay = MSD_REFRESH_DELAY;
00480         }
00481 
00482         /* Serial Timer */
00483         if (isSerialPortON)
00484         {
00485             /* Get Received size */
00486             size = DMAC_GetDestinationAddr(pDmad->pDmacs[iController], iChannel)
00487                     - (uint32_t)(usartBuffers[usartCurrentBuffer]);
00488             if (size == 0)
00489             {
00490                 pTc0->TC_CHANNEL[0].TC_CCR = TC_CCR_CLKEN | TC_CCR_SWTRG;
00491                 return;
00492             }
00493 
00494             /* Stop DMA */
00495             DMAD_StopTransfer(pDmad, dmaChannel);
00496             /* Restart DMA in next buffer */
00497             _UsartDmaRx( nextBuffer );
00498 
00499             /* Send current buffer through the USB */
00500             CDCDSerial_Write(usartBuffers[usartCurrentBuffer],
00501                              size, 0, 0);
00502             usartCurrentBuffer = nextBuffer;
00503             pTc0->TC_CHANNEL[0].TC_CCR = TC_CCR_CLKEN | TC_CCR_SWTRG;
00504         }
00505     }
00506 }
