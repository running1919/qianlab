00441  * Monitor keyboard buttons & Update key status in HID driver
00442  */
00443 static void HIDDKeyboardProcessKeys(void)
00444 {
00445     uint32_t i;
00446     uint8_t pressedKeys[NUM_KEYS];
00447     uint8_t pressedKeysSize = 0;
00448     uint8_t releasedKeys[NUM_KEYS];
00449     uint8_t releasedKeysSize = 0;
00450 
00451     /* Monitor buttons */
00452   #ifdef NO_PUSHBUTTON
00453     if (DBGU_IsRxReady())
00454     {
00455         uint8_t key = DBGU_GetChar();
00456         switch(key)
00457         {
00458             case '1': case '2':
00459                 i = key - '1';
00460                 if (keyStatus[i])
00461                 {
00462                     /* Key press simulation */
00463                     TRACE_INFO("Key %u pressed\n\r", i);
00464                     keyStatus[i] = 0;
00465                     pressedKeys[pressedKeysSize] = keyCodes[i];
00466                     pressedKeysSize ++;
00467                     HIDDKeyboard_RemoteWakeUp();
00468                 }
00469                 else
00470                 {
00471                     /* Key release simulation */
00472                     TRACE_INFO("Key %u released\n\r", i);
00473                     keyStatus[i] = 1;
00474                     releasedKeys[releasedKeysSize] = keyCodes[i];
00475                     releasedKeysSize++;
00476                 }
00477                 break;
00478             default: DBGU_PutChar(key);
00479         }
00480     }
00481   #else
00482     for (i=0; i < PIO_LISTSIZE(pinsPushButtons); i++)
00483     {
00484         /* Check if button state has changed */
00485         uint8_t isButtonPressed = PIO_Get(&(pinsPushButtons[i]));
00486         if (isButtonPressed != keyStatus[i])
00487         {
00488             /* Update button state */
00489             if (!isButtonPressed)
00490             {
00491                 /* Key has been pressed */
00492                 TRACE_INFO("Key %u has been pressed\n\r", i);
00493                 keyStatus[i] = 0;
00494                 pressedKeys[pressedKeysSize] = keyCodes[i];
00495                 pressedKeysSize++;
00496                 HIDDKeyboard_RemoteWakeUp();
00497             }
00498             else
00499             {
00500                 /* Key has been released */
00501                 TRACE_INFO("Key %u has been released\n\r", i);
00502                 keyStatus[i] = 1;
00503                 releasedKeys[releasedKeysSize] = keyCodes[i];
00504                 releasedKeysSize++;
00505             }
00506         }
00507     }
00508   #endif
00509 
00510     /* Update key status in the HID driver if necessary */
00511     if ((pressedKeysSize != 0) || (releasedKeysSize != 0))

00514 
00515         do
00516         {
00517             status = HIDDKeyboard_ChangeKeys(pressedKeys,
