00030 /** \addtogroup rtc_module Working with RTC
00031  * The RTC driver provides the interface to configure and use the RTC
00032  * peripheral.
00033  *
00034  * It manages date, time, and alarms.\n
00035  * This timer is clocked by the 32kHz system clock, and is not impacted by
00036  * power management settings (PMC). To be accurate, it is better to use an
00037  * external 32kHz crystal instead of the internal 32kHz RC.\n
00038  *
00039  * It uses BCD format, and time can be set in AM/PM or 24h mode through a
00040  * configuration bit in the mode register.\n
00041  *
00042  * To update date or time, the user has to follow these few steps :
00043  * <ul>
00044  * <li>Set UPDTIM and/or UPDCAL bit(s) in RTC_CR,</li>
00045  * <li>Polling or IRQ on the ACKUPD bit of RTC_CR,</li>
00046  * <li>Clear ACKUPD bit in RTC_SCCR,</li>
00047  * <li>Update Time and/or Calendar values in RTC_TIMR/RTC_CALR (BCD format),</li>
00048  * <li>Clear UPDTIM and/or UPDCAL bit in RTC_CR.</li>
00049  * </ul>
00050  * An alarm can be set to happen on month, date, hours, minutes or seconds,
00051  * by setting the proper "Enable" bit of each of these fields in the Time and
00052  * Calendar registers.
00053  * This allows a large number of configurations to be available for the user.
00054  * Alarm occurence can be detected even by polling or interrupt.
00055  *
00056  * A check of the validity of the date and time format and values written by the user is automatically done.
00057  * Errors are reported through the Valid Entry Register.
00058  *
00059  * For more accurate information, please look at the RTC section of the
00060  * Datasheet.
00061  *
00062  * Related files :\n
00063  * \ref rtc.c\n
00064  * \ref rtc.h.\n
00065 */

00067 /*@}*/

00149  *
00150  * \note In successive update operations, the user must wait at least one second
00151  * after resetting the UPDTIM/UPDCAL bit in the RTC_CR before setting these
00152  * bits again. Please look at the RTC section of the datasheet for detail.
00153  *
00154  * \param ucHour    Current hour in 12 or 24 hour mode.
00155  * \param ucMinute  Current minute.
00156  * \param ucSecond  Current second.
00157  *
00158  * \return 0 sucess, 1 fail to set
00159  */

00163     uint8_t ucHour_bcd ;
00164     uint8_t ucMin_bcd ;
00165     uint8_t ucSec_bcd ;
00166 
00167     TRACE_DEBUG( "RTC_SetTime(%02d:%02d:%02d)\n\r", ucHour, ucMinute, ucSecond ) ;
00168 
00169     /* if 12-hour mode, set AMPM bit */
00170     if ( (pRtc->RTC_MR & RTC_MR_HRMOD) == RTC_MR_HRMOD )
00171     {
00172         if ( ucHour > 12 )
00173         {
00174             ucHour -= 12 ;
00175             dwTime |= RTC_TIMR_AMPM ;
00176         }
00177     }
00178     ucHour_bcd = (ucHour%10)   | ((ucHour/10)<<4) ;
00179     ucMin_bcd  = (ucMinute%10) | ((ucMinute/10)<<4) ;
00180     ucSec_bcd  = (ucSecond%10) | ((ucSecond/10)<<4) ;
00181 
00182     /* value overflow */
00183     if ( (ucHour_bcd & (uint8_t)(~RTC_HOUR_BIT_LEN_MASK)) |
00184          (ucMin_bcd & (uint8_t)(~RTC_MIN_BIT_LEN_MASK)) |
00185          (ucSec_bcd & (uint8_t)(~RTC_SEC_BIT_LEN_MASK)))
00186     {
00187         return 1 ;
00188     }
00189 
00190     dwTime = ucSec_bcd | (ucMin_bcd << 8) | (ucHour_bcd<<16) ;
00191 
00192     pRtc->RTC_CR |= RTC_CR_UPDTIM ;
00193     while ((pRtc->RTC_SR & RTC_SR_ACKUPD) != RTC_SR_ACKUPD) ;
00194     pRtc->RTC_SCCR = RTC_SCCR_ACKCLR ;
00195     pRtc->RTC_TIMR = dwTime ;
00196     pRtc->RTC_CR &= (uint32_t)(~RTC_CR_UPDTIM) ;
