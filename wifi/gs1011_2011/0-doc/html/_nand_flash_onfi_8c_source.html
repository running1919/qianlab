00061 /**
00062  * \brief This function Reads the status register of the NAND device by 
00063  * issuing a 0x70 command.
00064  * \return  NAND_IO_RC_PASS     =0 : The function completes operation successfully.
00065             NAND_IO_RC_FAIL     =1 : The function does not complete operation successfully.
00066             NAND_IO_RC_TIMEOUT  =2 : The function times out before operation completes.
00067 */
00068 static uint32_t _NandReadStatus(void)
00069 {
00070     uint32_t nReadStatusCount;
00071     uint8_t ucStatus;
00072 
00073     /* Issue command */
00074     WRITE_NAND_COMMAND(NAND_CMD_STATUS, EBI_NF_ADDR);
00075     nReadStatusCount = 0;
00076 
00077     while (nReadStatusCount < MAX_READ_STATUS_COUNT) {
00078         /* Read status byte */
00079         ucStatus = READ_NAND(EBI_NF_ADDR);
00080         /* Check status */
00081         /* If status bit 6 = 1 device is ready */
00082         if ((ucStatus & STATUS_BIT_6) == STATUS_BIT_6) {
00083             if ((ucStatus & STATUS_BIT_0) == 0)  /* If status bit 0 = 0 the last operation was succesful */
00084                 return NAND_IO_RC_PASS;
00085             else
00086                 return NAND_IO_RC_FAIL;
00087         }
00088         nReadStatusCount++;
00089     }
00090     return NAND_IO_RC_TIMEOUT;
00091 }

00179 /**
00180  * \brief This function check if the Nandflash has an embedded ECC controller.
00181  * \return  0: ONFI not compliant or internal ECC not supported.
00182             1: Internal ECC enabled.
00183 */
00184 uint8_t NandEnableInternalEcc (void)
00185 {
00186     OnfiPageParam *pOnfiPageParameter;
00187     pOnfiPageParameter = NandGetCurrentOnfiInstance();
00188 
00189     if (pOnfiPageParameter->onfiCompatiable == 1) {
00190         /* Check if the Nandflash has an embedded ECC controller
00191            Known memories with this feature :
00192            - Manufacturer ID = 2Ch (Micron)
00193            - Number of bits ECC = 04h (4-bit ECC means process 34nm)
00194            - device size = 1Gb or 2Gb or 4Gb (Number of data bytes per page * Number of pages per block * Number of blocks per unit)  */
00195         if ( ((pOnfiPageParameter->manufacturerId & NAND_MFR_MICRON) == NAND_MFR_MICRON) &&
00196               (pOnfiPageParameter->onfiEccCorrectability == 0x4) &&
00197               ((pOnfiPageParameter->onfiDeviceModel == '1')     //  1G,
00198                || (pOnfiPageParameter->onfiDeviceModel == '2')     //  2G
00199                || (pOnfiPageParameter->onfiDeviceModel == '4'))) { //  or 4G bits
00200 
00201             /* then activate the internal ECC controller */
00202             WRITE_NAND_COMMAND(NAND_CMD_SET_FEATURE, EBI_NF_ADDR);
00203             WRITE_NAND_ADDRESS(0x90, EBI_NF_ADDR);
00204 
00205             WRITE_NAND(0x08, EBI_NF_ADDR);
00206             WRITE_NAND(0x00, EBI_NF_ADDR);
00207             WRITE_NAND(0x00, EBI_NF_ADDR);
00208             WRITE_NAND(0x00, EBI_NF_ADDR);
00209             NandSwitchEcc(ECC_INTERNAL);
00210             return 1;
00211         }
00212     }
00213     return 0;

00236     for (timer = 0; timer < 60; timer++) {
00237         rc = _NandReadStatus();
00238         if (rc == NAND_IO_RC_PASS) {
00239             WRITE_NAND_COMMAND(NAND_CMD_READID, EBI_NF_ADDR);
00240             WRITE_NAND_ADDRESS(0, EBI_NF_ADDR);

00250             break;
00251         }
00252     }
