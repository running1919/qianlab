00089 /**
00090  * \brief Configures a TWI peripheral to operate in master mode, at the given
00091  * frequency (in Hz). The duty cycle of the TWI clock is set to 50%.
00092  * \param pTwi  Pointer to an Twi instance.
00093  * \param twck  Desired TWI clock frequency.
00094  * \param mck  Master clock frequency.
00095  */
00096 void TWI_ConfigureMaster( Twi* pTwi, uint32_t dwTwCk, uint32_t dwMCk )
00097 {
00098     uint32_t dwCkDiv = 0 ;
00099     uint32_t dwClDiv ;
00100     uint32_t dwOk = 0 ;
00101 
00102     TRACE_DEBUG( "TWI_ConfigureMaster()\n\r" ) ;
00103     assert( pTwi ) ;
00104 
00105     /* SVEN: TWI Slave Mode Enabled */
00106     pTwi->TWI_CR = TWI_CR_SVEN ;
00107     /* Reset the TWI */
00108     pTwi->TWI_CR = TWI_CR_SWRST ;
00109     pTwi->TWI_RHR ;
00110 
00111     /* TWI Slave Mode Disabled, TWI Master Mode Disabled. */
00112     pTwi->TWI_CR = TWI_CR_SVDIS ;
00113     pTwi->TWI_CR = TWI_CR_MSDIS ;
00114 
00115     /* Set master mode */
00116     pTwi->TWI_CR = TWI_CR_MSEN ;
00117 
00118     /* Configure clock */
00119     while ( !dwOk )
00120     {
00121         dwClDiv = ((dwMCk / (2 * dwTwCk)) - 4) / (1<<dwCkDiv) ;
00122 
00123         if ( dwClDiv <= 255 )
00124         {
00125             dwOk = 1 ;
00126         }
00127         else
00128         {
00129             dwCkDiv++ ;
00130         }
00131     }
00132 
00133     assert( dwCkDiv < 8 ) ;
00134     TRACE_DEBUG( "Using CKDIV = %u and CLDIV/CHDIV = %u\n\r", dwCkDiv, dwClDiv ) ;
00135 
00136     pTwi->TWI_CWGR = 0 ;
00137     pTwi->TWI_CWGR = (dwCkDiv << 16) | (dwClDiv << 8) | dwClDiv ;
00138 }
