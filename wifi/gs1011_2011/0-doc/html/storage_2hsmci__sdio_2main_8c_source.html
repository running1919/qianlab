00268 /**
00269  * \brief Get 32-bit number input (Dec or Hex).
00270  * Before first character input format can be changed once by
00271  * 'x' to hex.
00272  * \param nbChar Number of character to wait.
00273  * \param pNum   Pointer to uint32_t for input result.

00277 {
00278     uint8_t key, isHex = 0;
00279     uint32_t  i;
00280     uint32_t  result = 0;
00281     for (i = 0; i < nbChar;) {
00282         key = DBGU_GetChar();
00283         /* User cancel input */
00284         if (key == 27) {
00285             printf(" Canceled\n\r");
00286             return key;
00287         }
00288         /* Short input */
00289         if (key == '\r') break;
00290         if (key == 'x' && i == 0) {
00291             if (isHex == 0) {
00292                 isHex = 1;
00293                 DBGU_PutChar(key);
00294             }
00295             continue;
00296         }
00297         if (key > '9' || key < '0') {
00298             if (isHex) {
00299                 if (key < 'a' || key > 'z') {
00300                     continue;
00301                 }
00302             }
00303             else {
00304                 continue;
00305             }
00306         }
00307         DBGU_PutChar(key);
00308         if (isHex) {
00309             if (key >= 'a')
00310                 result = result * 16 + (key - 'a' + 10);
00311             else
00312                 result = result * 16 + (key - '0');
00313         }
00314         else {
00315             result = result * 10 + (key - '0');
00316         }
00317         i ++;
00318     }
00319     if (pU32) *pU32 = result;
00320     return 0;
